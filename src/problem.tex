\section{基本概念与运算}

\subsection{基本概念}

\textbf{实验 v.s. 试验}（《现代汉语词典》定义）
\begin{itemize}[leftmargin=2em]
    \item \textbf{实验}：为了检验某种科学理论或假设而进行某种操作或从事某种活动。
    \item \textbf{试验}：为了察看某事的结果或某物的性能而从事某种活动。
\end{itemize}

\textbf{随机试验}$E$是满足如下三个特征的试验：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{可重复性}：可在相同条件下重复进行；
    \item \textbf{不确定性}：一次试验之前无法确定具体出现哪种结果；
    \item \textbf{完备性}：能确定所有的可能结果。
\end{enumerate}

\textbf{随机试验}$\to$\textbf{随机事件}$\to$\textbf{概率}

\begin{itemize}[leftmargin=2em]
    \item \textbf{样本点}：随机试验的单个结果。
    \item \textbf{样本空间}$S$：随机试验所有可能的结果组成的集合。
    \item \textbf{随机事件}：样本空间的任意一个子集。
    \item \textbf{基本事件}：有一个样本点组成的单点集。
    \item \textbf{必然事件}$S$：样本空间中所有样本点构成的集合。
    \item \textbf{不可能事件}$\varnothing$：不包含任何样本点的空集。
\end{itemize}

\begin{remark}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 随机事件是定义在随机试验基础上的；概率是随机事件的概率。
    \item 随机事件的"集合"定义形式，促成了事件的运算与集合运算的关系。
\end{enumerate}
\end{remark}

\subsection{事件之间的运算规则}

\begin{itemize}[leftmargin=2em]
    \item \textbf{交换律}：$A\cup B=B\cup A$，$A\cap B=B\cap A$
    \item \textbf{结合律}：
    \[
    A\cup(B\cup C)=(A\cup B)\cup C,\quad A\cap(B\cap C)=(A\cap B)\cap C
    \]
    \item \textbf{分配律}：
    \[
    A\cup(B\cap C)=(A\cup B)\cap(A\cup C),\quad A\cap(B\cup C)=(A\cap B)\cup(A\cap C)
    \]
    \item \textbf{De Morgan律}：
    \[
    \overline{A\cup B}=\bar{A}\cap\bar{B},\quad \overline{A\cap B}=\bar{A}\cup\bar{B}
    \]
\end{itemize}

\begin{example}[事件运算实例]
\begin{itemize}[leftmargin=2em]
    \item "$A,B,C$中至少有一个发生"：$A\cup B\cup C$
    \item "$A,B,C$中至少有两个发生"：$AB\cup BC\cup AC$
    \item "$A,B,C$中最多有一发生"：
    \[
    A\bar{B}\bar{C}+\bar{A}B\bar{C}+\bar{A}\bar{B}C+\bar{A}\bar{B}\bar{C}=\overline{AB\cup BC\cup AC}
    \]
\end{itemize}
\end{example}

\begin{example}[De Morgan律证明]
证明$\overline{A\cup B}=\bar{A}\cap\bar{B}$。

\textbf{解}：
\[
\begin{aligned}
\forall x\in\overline{A\cup B}&\Rightarrow x\not\in A\cup B\\
&\Rightarrow x\not\in A\text{ 且 }x\not\in B\\
&\Rightarrow x\in\bar{A}\text{ 且 }x\in\bar{B}\\
&\Rightarrow x\in\bar{A}\cap\bar{B}\\
&\Rightarrow\overline{A\cup B}\subseteq\bar{A}\cap\bar{B}.
\end{aligned}
\]
\[
\begin{aligned}
\forall x\in\bar{A}\cap\bar{B}&\Rightarrow x\in\bar{A}\text{ 且 }x\in\bar{B}\\
&\Rightarrow x\not\in A\text{ 且 }x\not\in B\\
&\Rightarrow x\not\in A\cup B\\
&\Rightarrow x\in\overline{A\cup B}\\
&\Rightarrow\bar{A}\cap\bar{B}\subseteq\overline{A\cup B}.
\end{aligned}
\]
\end{example}

\section{频率与概率}

\textbf{频率}：描述事件发生的频繁程度。

\textbf{概率}：表征随机事件在一次试验中发生的可能性大小。

\subsection{频率与概率的定义}

事件$A$发生的\textbf{频率}记为$f_n(A)=\dfrac{n_A}{n}$：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $0\leq f_n(A)\leq1$
    \item $f_n(\Omega)=1$
    \item 若$A_1,A_2,\ldots,A_k$两两互不相容事件，则
    \[
    f_n(A_1\cup A_2\cup\cdots\cup A_k)=f_n(A_1)+f_n(A_2)+\cdots+f_n(A_k)
    \]
\end{enumerate}

事件$A$发生的\textbf{概率}记为$P(A)$：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $P(A)\geq0$；非负性
    \item $P(\Omega)=1$；规范性
    \item 若$A_1,A_2,\ldots$两两互不相容事件，即$A_iA_j=\varnothing$，$i\neq j$，$i,j=1,2,\ldots$，则
    \[
    P(A_1\cup A_2\cup\cdots)=P(A_1)+P(A_2)+\cdots\quad\text{（可列可加性）}
    \]
\end{enumerate}

\textbf{频率的稳定性}：$n\to\infty$，$f_n(A)\to P(A)$。

\subsection{概率的性质与推广}

\textbf{性质1}：$P(\varnothing)=0$

\textbf{性质2}：$P(A)\leq1$

\textbf{性质3}：若$A_1,A_2,\ldots,A_n$是两两不相容事件，则
\[
P(A_1\cup A_2\cup\cdots\cup A_n)=P(A_1)+P(A_2)+\cdots+P(A_n)\quad\text{（有限可加性）}
\]

\textbf{性质4}（包含可减性）：$A\subset B\Rightarrow P(B-A)=P(B)-P(A)$

\textbf{性质5}（逆事件的概率公式）：$P(\bar{A})=1-P(A)$

\textbf{性质6}：$P(A\cup B)=P(A)+P(B)-P(AB)$

\textbf{一般推广}：
\[
P(B\bar{A})=P(B-A)=P(B)-P(AB)
\]
\[
P(\bar{A}\bar{B}\bar{C})=P(\overline{A\cup B\cup C})=1-P(A\cup B\cup C)
\]

\textbf{加法公式推广}：
\[
P(A_1\cup A_2\cup\cdots\cup A_n)=\sum_{i=1}^{n}P(A_i)-\sum_{1\leq i<j\leq n}P(A_iA_j)+\sum_{1\leq i<j<k\leq n}P(A_iA_jA_k)+\cdots+(-1)^{n-1}P(A_1A_2\ldots A_n)
\]

\begin{example}[加法公式证明]
证明$P(A\cup B\cup C)=P(A)+P(B)+P(C)-P(AB)-P(AC)-P(BC)+P(ABC)$。

\textbf{解}：
\[
A\cup B\cup C=A\cup(B-AB)\cup(C-C(A\cup B))
\]
因为$A$、$B-AB$、$C-C(A\cup B)$两两互斥，所以
\[
\begin{aligned}
P(A\cup B\cup C)&=P(A)+P(B-AB)+P(C-C(A\cup B))\\
&=P(A)+P(B)-P(AB)+P(C)-P(C(A\cup B))\\
&=P(A)+P(B)-P(AB)+P(C)-P(AC\cup BC)\\
&=P(A)+P(B)-P(AB)+P(C)-P(AC)-P(BC)+P(ABC)\\
&=P(A)+P(B)+P(C)-P(AB)-P(AC)-P(BC)+P(ABC).
\end{aligned}
\]
\end{example}

\section{古典概型与几何概型}

\subsection{古典概型（等可能概型）}

\textbf{特点}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 样本空间的元素只有有限个；（有限性）
    \item 每个基本事件发生的可能性相同。（等可能性）
\end{enumerate}

\textbf{随机事件的概率}：
\[
P(A)=\frac{A\text{ 中元素个数}}{\Omega\text{ 中元素个数}}=\frac{N(A)}{N(\Omega)}
\]

\textbf{几个经典问题}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{无放回抽样}：超几何分布，$p_k=\dfrac{C_D^kC_{N-D}^{n-k}}{C_N^n}$
    \item \textbf{有放回抽样}：二项分布，$p_k=\dfrac{C_n^kD^k(N-D)^{n-k}}{N^n}$
    \item \textbf{生日问题}：$n$个人生日各不相同，$p=\dfrac{365!}{365^n(365-n)!}$
    \item \textbf{分组数量}：$n$个球分$m$组，第$i$组有$n_i$个球，一共有$\dfrac{n!}{n_1!n_2!\cdots n_m!}$种分法
    \item \textbf{有放回与无放回抽签}：$\dfrac{a}{a+b}$
\end{enumerate}

\subsection{几何概型}

古典概型只考虑了有限等可能的随机试验，几何概型考虑了有无穷多个等可能结果的随机试验。

\textbf{几何概型的随机试验特点}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 样本空间$\Omega$是一个几何区域；
    \item 随机试验$E$相当于在区域内任意地取点，且取到每一点都是等可能的。
\end{enumerate}

如果试验$E$是向区域内任意取点，事件$A$对应于点落在$\Omega$内的某区域$A$，则
\[
P(A)=\frac{m_A}{m_\Omega}
\]
其中，$m_\Omega$、$m_A$是可度量区域$\Omega$、$A$的度量（如线段长度、平面区域面积、空间区域的体积等）。

\begin{example}[会面问题]
甲、乙二人约定在5点到12点之间在某地会面，先到者等一个小时后即离去。设二人在这段时间内的各时刻到达是等可能的，且二人互不影响。求二人能会面的概率。

\section{条件概率与Bayes公式}

\subsection{条件概率}

\textbf{条件概率}：在事件$A$发生的条件下事件$B$发生的概率。

\textbf{公式法}：
\[
P(B\mid A)=\frac{P(AB)}{P(A)},\quad\text{其中 }P(A)>0
\]

\textbf{缩小样本空间法}——适用于古典概型

设事件$A$所含样本点数为$n_A$，事件$AB$所含样本点数为$n_{AB}$，则
\[
P(B\mid A)=\frac{n_{AB}}{n_A}
\]

\begin{remark}
条件概率与无条件概率的大小无确定关系，$P(B)$与$P(B\mid A)$无确定的大小关系。
\end{remark}

\subsection{乘法公式和全概率公式}

\textbf{乘法公式}：
\[
P(AB)=P(B\mid A)P(A),\quad P(A)>0
\]

$n$个事件：假设$P(A_1A_2\ldots A_{n-1})>0$，有：
\[
P(A_1A_2\ldots A_n)=P(A_n\mid A_1A_2\ldots A_{n-1})\cdot P(A_{n-1}\mid A_1A_2\ldots A_{n-2})\cdots P(A_2\mid A_1)P(A_1)
\]

\textbf{全概率公式}

\textbf{划分}："交为空，并为全"

$B_1,B_2,\ldots,B_n$是样本空间$\Omega$的一个划分
\[
P(A)=\sum_{i=1}^{n}P(AB_i)=\sum_{i=1}^{n}P(B_i)P(A\mid B_i)
\]
（已知原因/因素，求结果）

\subsection{Bayes公式}

\begin{figure}[htbp]
    \centering
    \includegraphics[width=0.5\textwidth]{figures/fig_bayes_diagram.jpg}
    \caption{Bayes公式示意图（先验、似然、后验的关系）}
    \label{fig:bayes_diagram}
\end{figure}

\[
P(B_k\mid A)=\frac{P(AB_k)}{P(A)}=\frac{P(A\mid B_k)P(B_k)}{P(A)}=\frac{P(B_k)P(A\mid B_k)}{\sum_{i=1}^{n}P(A\mid B_i)P(B_i)}
\]

\textbf{理解1}：【执果溯因】已知结果，求原因的可能性。

\textbf{理解2}：【经验更新】先验概率，结合新获得的信息，得到修正后的结论。

\begin{example}[拼写检查]
背景材料：一本字典$D$，一个语料库$S$。

问题：用户输入了一个单词$w$，判断此单词是否正确（即是否可在字典$D$中查询到），如果$w$不正确，给出可能的正确单词拼写。

如果$w$不在字典$D$中，则$w$拼写错误。

为了找出最可能正确的单词，考察字典中与$w$形近的单词$c$：

$P(c)$：单词$c$出现的先验概率，可从语料库$S$中统计$c$出现的频率。

通过Bayes公式利用似然函数$P(w\mid c)$更新概率：
\[
P(c\mid w)=\frac{P(w\mid c)P(c)}{P(w)}\propto P(w\mid c)P(c)
\]

$P(w\mid c)$：在试图拼写$c$的情况下，出现拼写错误$w$的概率。可根据$w$与$c$的距离计算：
\begin{itemize}[leftmargin=2em]
    \item 莱文斯坦距离（Levenshtein distance）
    \item 最长公共子串长度
\end{itemize}
\end{example}

\section{独立性}

\subsection{两事件独立的定义}

\[
P(AB)=P(A)P(B)
\]

\subsection{两事件独立性的性质}

\begin{itemize}[leftmargin=2em]
    \item $P(A)>0$，事件$A$与$B$相互独立的充分必要条件：$P(B\mid A)=P(B)$
    \item \textbf{逆事件的独立性}：若随机事件$A$与$B$相互独立，则$\bar{A}$与$B$，$A$与$\bar{B}$，$\bar{A}$与$\bar{B}$也相互独立
    \item 必然事件$S$与任意随机事件$A$相互独立
    \item 不可能事件$\varnothing$与任意随机事件$A$相互独立
\end{itemize}

\begin{remark}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{两事件相互独立 vs 互不相容}
    \begin{itemize}
        \item "$A$与$B$互不相容"指两事件不能同时发生，即$AB=\varnothing$。（$P(AB)=0\nRightarrow AB$互不相容）
        \item "$A$与$B$相互独立"指$A$是否发生不影响$B$发生的概率
    \end{itemize}
    \item 设事件$A$与$B$满足$P(A)P(B)\neq0$，则互不相容与相互独立不能同时成立，即
    \begin{itemize}
        \item 若事件$A$与$B$相互独立，则$AB\neq\varnothing$
        \item 若$AB=\varnothing$，则事件$A$与$B$不相互独立
    \end{itemize}
    \item 区分"对立事件"与"相互独立事件"：对立事件不相互独立。
\end{enumerate}
\end{remark}

\subsection{三个事件的独立性}

设$A$、$B$、$C$是三个随机事件，如果
\[
\begin{cases}
P(AB)=P(A)P(B)\\
P(AC)=P(A)P(C)\\
P(BC)=P(B)P(C)\\
P(ABC)=P(A)P(B)P(C)
\end{cases}
\]
四个等式缺一不可，则称$A$、$B$、$C$是相互独立的随机事件。

\begin{remark}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 在三个事件独立性的定义中，四个等式缺一不可。即前三个等式的成立不能推出第四等式的成立；反之，最后一个等式的成立也推不出前三个等式的成立。
    \item 若$A,B,C$是相互独立的三个事件，则通过事件$B$、$C$运算得到的事件也与事件$A$相互独立，例如$(A,\overline{B\cup C})$，$(A,\overline{BC})$，$(A,\overline{B}-\overline{C})$，$(A,\overline{B\cup C})$，$(A,\overline{BC})$，$(A,\bar{B}\bar{C})$等。
\end{enumerate}
\end{remark}

\subsection{\texorpdfstring{$n$}{n}个事件的相互独立性}

设$A_1,A_2,\ldots,A_n$为$n$个随机事件，若下列等式成立
\[
\begin{cases}
P(A_iA_j)=P(A_i)P(A_j) & (1\leq i<j\leq n)\\
P(A_iA_jA_k)=P(A_i)P(A_j)P(A_k) & (1\leq i<j<k\leq n)\\
\cdots\cdots & \cdots\\
P(A_1A_2\ldots A_n)=P(A_1)P(A_2)\cdots P(A_n)
\end{cases}
\]
则称$A_1,A_2,\ldots,A_n$这$n$个随机事件相互独立。

\section{Bernoulli概型}

\textbf{Bernoulli试验}：若随机试验$E$只有两个结果。一般地，我们将这两个结果记为$A$与$\bar{A}$，分别称为"成功"与"失败"。

\textbf{$n$重Bernoulli试验}：独立重复地进行$n$次Bernoulli试验
\begin{itemize}[leftmargin=2em]
    \item "重复"：指每次试验中$A$发生的概率（即"成功"的概率）不变
    \item "独立"：指各次试验的结果相互独立
\end{itemize}

\textbf{$n$重Bernoulli试验中恰好成功$k$次的概率}：

设在一次Bernoulli试验中，$P(A)=p$，$P(\bar{A})=1-p=q$。

事件$B_{n,k}=\{n\text{ 重Bernoulli试验中事件}A\text{ 恰好发生}k\text{ 次}\}$的概率：
\[
P(B_{n,k})=C_n^kp^k(1-p)^{n-k},\quad k=0,1,2,\ldots,n
\]

\begin{example}[《概率论及其应用》1.8 \#14]
证明$\overline{A\cup B}=\bar{A}\cap\bar{B}$。

多种证明方式：根据定义，画图，利用公式等。本题的目的是考察对定义的了解，所以推荐使用定义证明。

\textbf{解}：
\[
\begin{aligned}
\forall x\in\overline{A\cup B}&\Rightarrow x\not\in A\cup B\\
&\Rightarrow x\not\in A\text{ 且 }x\not\in B\\
&\Rightarrow x\in\bar{A}\text{ 且 }x\in\bar{B}\\
&\Rightarrow x\in\bar{A}\cap\bar{B}\\
&\Rightarrow\overline{A\cup B}\subseteq\bar{A}\cap\bar{B}.
\end{aligned}
\]
\[
\begin{aligned}
\forall x\in\bar{A}\cap\bar{B}&\Rightarrow x\in\bar{A}\text{ 且 }x\in\bar{B}\\
&\Rightarrow x\not\in A\text{ 且 }x\not\in B\\
&\Rightarrow x\not\in A\cup B\\
&\Rightarrow x\in\overline{A\cup B}\\
&\Rightarrow\bar{A}\cap\bar{B}\subseteq\overline{A\cup B}.
\end{aligned}
\]
\end{example}

\begin{example}[《概率论及其应用》2.10 \#9]
计算：$n$个球随机放入$n$个盒，恰有一盒空着的概率。

\textbf{解}：
\begin{itemize}[leftmargin=2em]
    \item 总的放法个数：$n^n$
    \item 空盒的可能个数：$C_n^1$
    \item 问题变为：$n$个球装入$n-1$个盒子，不能有空
    \item 从而某个盒子必须有2个球，两个球的构成方式：$C_n^2$
    \item 放入盒子的方式：$(n-1)!$
    \item 结果为：
    \[
    \frac{C_n^1\cdot C_n^2\cdot(n-1)!}{n^n}=\frac{C_n^2\cdot n!}{n^n}
    \]
\end{itemize}
\end{example}

\begin{example}[《概率论及其应用》2.10 \#21]
谣言的传播：在一个拥有$n+1$个居民的城市里，某人告诉第二个人一个谣言，第二个人又把谣言告诉第三个人，如此等等。在每一步中，谣言的接收者都是随机地从$n$个居民中挑选。

求下述两事件的概率：谣言传播$r$次后，
\begin{enumerate}[label=(\alph*)]
    \item 还没有回到第一个造谣者；
    \item 没有一个人两次听到谣言。
\end{enumerate}

当每一次都把谣言同时告诉由城市中随机选取的$N$个居民时，(a)(b)两事件的概率等于多少？

\textbf{解}：$N=1$时，共有$n^r$种等可能传播情况
\begin{enumerate}[label=(\alph*)]
    \item 还没有回到第一个造谣者的情况有$n(n-1)^{r-1}$种
    \[
    P_A=\frac{n(n-1)^{r-1}}{n^r}=\left(1-\frac{1}{n}\right)^{r-1}
    \]
    \item 没有一人两次听到谣言的情况有$A_n^r=\dfrac{n!}{(n-r)!}$种
    \[
    P_B=\frac{A_n^r}{n^r}
    \]
\end{enumerate}

\textbf{解（续）}：若每次传播随机选取$N$个居民，共有$(C_n^N)^r$种等可能情况
\begin{enumerate}[label=(\alph*)]
    \item 没有回到第一个造谣者的情况有$C_n^N(C_{n-1}^N)^{r-1}$种
    \[
    P_A=\frac{C_n^N(C_{n-1}^N)^{r-1}}{(C_n^N)^r}=\left(1-\frac{N}{n}\right)^{r-1}
    \]
    \item 没有一人两次听到谣言的情况有$C_n^NC_{n-N}^NC_{n-2N}^N\cdots C_{n-(r-1)N}^N$种
    \[
    P_B=\frac{C_n^NC_{n-N}^NC_{n-2N}^N\cdots C_{n-(r-1)N}^N}{(C_n^N)^r}=\frac{n!}{(N!)^r(n-rN)!}
    \]
\end{enumerate}
\end{example}

\begin{example}[《概率论及其应用》2.10 \#24(b)]
求6个人的生日恰巧在两个月中的概率。（假定任何一个人生于12个月中之任一月都是等概率的）

\textbf{解}：
\[
\frac{C_{12}^2(2^6-2)}{12^6}
\]
本题结果中分子的乘子容易写成$2^6$而忘记减2，2表示6个人全部在其中的一个月生日的事件数。
\end{example}

\begin{example}[《概率论及其应用》2.10 \#38]
印刷错误：假设书中每一页都有$N$个印刷符号，他们有可能印错。全书共有$n=500$页，$r=50$个印错符号。证明：
\begin{enumerate}[label=(\alph*)]
    \item 第$1,2,\ldots,n$页分别含有$r_1,r_2,\ldots,r_n$个印错符号的概率为
    \[
    \frac{C_N^{r_1}C_N^{r_2}\cdots C_N^{r_n}}{C_{nN}^r}
    \]
    \item 当$N$充分大后，
    \[
    \frac{C_N^{r_1}C_N^{r_2}\cdots C_N^{r_n}}{C_{nN}^r}
    \]
    逼近$\dfrac{r!}{r_1!r_2!\cdots r_n!n^r}$。$r$个错误分布在$n$页的问题近似于$r$个球随机放入$n$个盒子的问题。
\end{enumerate}

\textbf{提示}：$N\to+\infty$，$N\gg c\Longrightarrow\dfrac{N!}{(N-c)!}\sim N^c$
\end{example}

\begin{example}[《概率论与数理统计》1.8（2）]
求至少有2件次品的概率。

\textbf{解}：

\end{example}

\begin{example}[《概率论与数理统计》1.12]
50只铆钉随机地取来用在10个部件上，其中有3只铆钉强度太弱。每个部件用3只铆钉。若将3只强度太弱的铆钉都钉在一个部件上，则这个部件强度就太弱。问发生一个部件强度太弱的概率是多少？

\textbf{解法1}：

50只铆钉随机用在10个部件上的可能情况：$\dfrac{A_{50}^{30}}{(3!)^{10}}$

一个部件强度为弱的可能情况：$C_{10}^1\dfrac{A_{47}^{27}}{(3!)^9}$

发生一个部件强度太弱的概率：
\[
P=\frac{C_{10}^1\dfrac{A_{47}^{27}}{(3!)^9}}{\dfrac{A_{50}^{30}}{(3!)^{10}}}=\frac{1}{1960}
\]

\textbf{解法2}：令$A_i$表示事件"第$i$号部件强度太弱"

从50只铆钉中取出3只放在第$i$号部件上的可能取法：$C_{50}^3$

强度太弱的铆钉都装在第$i$号部件上的可能取法：$C_3^3=1$

\[
P(A_i)=\frac{1}{C_{50}^3}=\frac{1}{19600}
\]

由于$A_1,A_2,\ldots,A_{10}$两两互不相容，
\[
P(A_1\cup A_2\cup\cdots\cup A_{10})=P(A_1)+P(A_2)+\cdots+P(A_{10})=\frac{1}{1960}
\]
\end{example}

\section{典型例题}

\begin{example}[例1]
已知$A$、$B$、$C$是三个两两独立的事件，且$ABC=\varnothing$，$P(A\cup B\cup C)=\dfrac{9}{16}$，$P(A)=P(B)=P(C)$，求$P(A)$。

\textbf{解}：
\[
\begin{aligned}
\frac{9}{16}&=P(A\cup B\cup C)\\
&=P(A)+P(B)+P(C)-P(AB)-P(AC)-P(BC)+P(ABC)\\
&=3P(A)-3P(A)^2
\end{aligned}
\]
求解得$P(A)=\dfrac{3}{4}$或$P(A)=\dfrac{1}{4}$。

又因为$P(A)<P(A\cup B\cup C)=\dfrac{9}{16}$，故$P(A)=\dfrac{1}{4}$。
\end{example}

\begin{example}[例2]
$A$、$B$是两个随机事件，$P(A\cup B)=0.7$，$P(A)=0.4$。
\begin{enumerate}[label=(\arabic*)]
    \item 若$A,B$互不相容，求$P(B)$；
    \item 若$A,B$相互独立，求$P(B)$；
    \item 若$P(B\mid A)=0.6$，求$P(B)$。
\end{enumerate}

\textbf{解}：$P(A\cup B)=P(A)+P(B)-P(AB)$

(1) $P(A\cup B)=P(A)+P(B)=0.7$，$P(B)=0.7-0.4=0.3$

(2) $P(AB)=P(A)P(B)$
\[
P(B)=\frac{P(A\cup B)-P(A)}{P(\bar{A})}=\frac{0.7-0.4}{0.6}=0.5
\]

(3) $P(AB)=P(A)P(B\mid A)$
\[
P(B)=P(A\cup B)-P(A)+P(A)P(B\mid A)=0.7-0.4+0.4\times0.6=0.5
\]
\end{example}

\begin{example}[例3]
在长为1的线段上任取两点，分成3个线段，求这些线段能构成三角形的概率。

y<x+(1-x-y)<1\\
x<y+(1-x-y)<1
\end{cases}
\Rightarrow
\begin{cases}
x+y>0.5\\
y<0.5\\
x<0.5
\end{cases}
\]
即落在图中阴影区域内。

\begin{example}[例4]
从混有5张假钞的20张百元钞票中任意抽出2张，将其中1张放到验钞机上检验发现是假钞。求2张都是假钞的概率。

\textbf{解}：令事件$A$表示抽到2张都是假钞，事件$B$表示2张中至少有1张假钞。

则所求概率是$P(A\mid B)$（而不是$P(A)$！）

\[
P(A\mid B)=\frac{P(AB)}{P(B)}=\frac{P(A)}{P(B)}=\frac{C_5^2/C_{20}^2}{(C_5^2+C_5^1C_{15}^1)/C_{20}^2}=\frac{C_5^2}{C_5^2+C_5^1C_{15}^1}=\frac{10}{85}=0.118
\]
\end{example}

\begin{example}[例5]
对以往的数据分析结果表明当机器调整得良好时，产品的合格率为90\%，而当机器发生某一故障时，其合格率为30\%。每天早上机器开动时，机器调整良好的概率为75\%。已知某天早上第一件产品是合格品，试求机器调整得良好的概率是多少？

\textbf{解}：设事件$A$：产品合格；事件$B$：机器调整良好，且$P(A\mid B)=0.9$；事件$\bar{B}$：机器发生故障，且$P(A\mid\bar{B})=0.3$。

\[
P(B\mid A)=\frac{P(A\mid B)P(B)}{P(A\mid B)P(B)+P(A\mid\bar{B})P(\bar{B})}=\frac{0.9\times0.75}{0.9\times0.75+0.3\times0.25}=0.9
\]
\end{example}

\section{随机变量}

\begin{definition}[随机变量]
设随机试验的样本空间是\(S=\{\omega\}\)，\(X=X(\omega)\)是定义在样本空间\(S\)上的实值单值函数，称\(X=X(\omega)\)为\textbf{随机变量}。
\end{definition}

\begin{remark}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{随机变量 v.s. 普通函数}：随机变量是一个定义在样本空间上的函数；普通函数定义在实数轴上。
    \item \textbf{随机变量的取值具有一定的概率规律}。
    \item \textbf{随机变量 v.s. 随机事件}：随机变量从动态的观点研究随机现象；随机事件从静态的观点研究随机现象。
\end{enumerate}
\end{remark}

\section{离散型随机变量的分布律}

\begin{definition}
设离散型随机变量\(X\)所有可能取的值为\(x_k\)（\(k=1,2,\cdots\)），\(X\)取各个可能值的概率，即事件\(\{X=x_k\}\)的概率，为
\[
P\{X=x_k\}=p_k,\quad k=1,2,\cdots
\]
称此为离散型随机变量\(X\)的\textbf{分布律}。
\end{definition}

\textbf{性质}：
\begin{itemize}[leftmargin=2em]
    \item \textbf{非负性}：\(p_k\geq 0\)，\(k=1,2,\cdots\)
    \item \textbf{归一性}：\(\sum_{k=1}^{\infty}p_k=1\)
\end{itemize}

分布律形式上可记为
\[
\begin{array}{c|ccccc}
X & x_1 & x_2 & \cdots & x_k & \cdots \\ \hline
p_k & p_1 & p_2 & \cdots & p_k & \cdots
\end{array}
\]

\section{常用的离散型随机变量}

\subsubsection{Bernoulli分布（两点分布）}

如果随机变量\(X\)的分布律为
\[
P\{X=1\}=p,\quad P\{X=0\}=1-p=q
\]
或
\[
\begin{array}{c|cc}
X & 0 & 1 \\ \hline
P & 1-p & p
\end{array}
\]
则称随机变量\(X\)服从参数为\(p\)的\textbf{Bernoulli分布}。

记作\(X\sim B(1,p)\)，其中\(0\leq p\leq 1\)为参数。

\textbf{概率背景}：进行一次Bernoulli试验，事件\(A\)是否发生。

\subsubsection{二项分布}

如果随机变量\(X\)的分布律为
\[
P\{X=k\}=C_n^k p^k(1-p)^{n-k},\quad k=0,1,2,\cdots,n
\]
则称随机变量\(X\)服从参数为\((n,p)\)的\textbf{二项分布}。

记作\(X\sim B(n,p)\)，其中\(0\leq p\leq 1\)为参数。

\textbf{概率背景}：进行\(n\)次Bernoulli试验，事件\(A\)发生的次数。

    \centering
    \includegraphics[width=0.5\textwidth]{figures/fig_binomial_pmf_exercise.png}
    \caption{二项分布概率质量函数（$n=100,p=0.5$）}
    \label{fig:binomial_pmf_exercise}
\end{figure}

\subsubsection{Poisson分布}

如果随机变量\(X\)的分布律为
\[
P\{X=k\}=\frac{\lambda^k}{k!}e^{-\lambda},\quad k=0,1,2,\cdots
\]
其中\(\lambda>0\)为常数，则称随机变量\(X\)服从参数为\(\lambda\)的\textbf{Poisson分布}。

记作\(X\sim P(\lambda)\)。

\textbf{常用场合}：\(X\)表示一定的时间或空间内出现的事件次数。

\begin{figure}[htbp]
    \centering
    \includegraphics[width=0.5\textwidth]{figures/fig_poisson_pmf_exercise.png}
    \caption{Poisson分布概率质量函数（$\lambda=50$）}
    \label{fig:poisson_pmf_exercise}
\end{figure}

\begin{theorem}[Poisson定理]
设随机变量\(X_n\)（\(n=1,2,\cdots\)）服从二项分布，其分布律为
\[
P\{X_n=k\}=C_n^k p_n^k(1-p_n)^{n-k},\quad k=0,1,\cdots,n
\]
若\(\lim_{n\to\infty}np_n=\lambda>0\)，则
\[
\lim_{n\to\infty}P\{X_n=k\}=\lim_{n\to\infty}C_n^k p_n^k(1-p_n)^{n-k}=\frac{\lambda^k}{k!}e^{-\lambda}
\]
\end{theorem}

\textbf{Poisson定理的应用}：
\begin{itemize}[leftmargin=2em]
    \item \(np\)较小时，用Poisson分布近似二项分布进行计算。（\(n\geq 20\)，\(p\leq 0.05\)时较好）
    \item 当\(n\)比较大，\(p\)比较小时，令\(\lambda=np\)，则
    \[
    P\{X=k\}=C_n^k p^k(1-p)^{n-k}\approx\frac{\lambda^k}{k!}e^{-\lambda}
    \]
\end{itemize}

\begin{figure}[htbp]
    \centering
    \includegraphics[width=0.5\textwidth]{figures/fig_poisson_approx.png}
    \caption{Poisson定理近似效果图}
    \label{fig:poisson_approx}
\end{figure}

\subsubsection{几何分布}

如果随机变量\(X\)的分布律为
\[

\]
其中\(p\geq 0\)，\(q\geq 0\)，\(p+q=1\)，则称随机变量\(X\)服从参数为\(p\)的\textbf{几何分布}。

\textbf{概率背景}：在Bernoulli试验中进行到事件\(A\)首次发生为止。

\]
其中\(M,N,n\)均为自然数，则称随机变量\(X\)服从参数为\((M,N,n)\)的\textbf{超几何分布}。

\section{随机变量的分布函数}

\begin{definition}
设\(X\)是一个随机变量，\(x\)是任意实数，函数
\[
F(x)=P\{X\leq x\}
\]
称为\(X\)的\textbf{分布函数}。
\end{definition}

\textbf{说明}：分布函数\(F(x)\)是\(x\)的一个普通实函数。分布函数主要研究随机变量在某一区间内取值的概率情况。

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{取值范围}：\(0\leq F(x)\leq 1\)，\((-\infty,\infty)\)
    \item \textbf{不减性}：若\(x_1<x_2\)，则\(F(x_1)\leq F(x_2)\)
    \item \textbf{右连续}：\(F(x+0)=F(x)\)
    \item \textbf{左右极限值}：
    \[
    F(-\infty)=\lim_{x\to-\infty}F(x)=0,\quad F(+\infty)=\lim_{x\to+\infty}F(x)=1
    \]
    \item \textbf{单点概率值计算}：
    \[
    P\{X=x_0\}=F(x_0)-F(x_0-0)
    \]
\end{enumerate}

具备前三条性质的函数\(F(x)\)必是某个随机变量的分布函数。

\textbf{重要计算公式}：
\begin{itemize}[leftmargin=2em]
    \item 利用分布函数计算区间概率：
    \[
    P\{a<X\leq b\}=F(b)-F(a)
    \]
    \[
    P\{X>a\}=1-F(a)
    \]
    \[
    P\{X=x_0\}=F(x_0)-F(x_0-0)
    \]
    \item 离散型随机变量的分布函数：
    \[
    F(x)=P\{X\leq x\}=\sum_{x_k\leq x}p_k
    \]
\end{itemize}

\section{连续型随机变量的概率密度}

\begin{definition}
如果对于随机变量\(X\)的分布函数\(F(x)\)，存在非负函数\(f(x)\)，使对于任意实数\(x\)有
\[
F(x)=\int_{-\infty}^{x}f(t)\,\mathrm{d}t
\]
称\(X\)为\textbf{连续型随机变量}，其中\(f(x)\)称为\(X\)的\textbf{概率密度函数}，简称\textbf{概率密度}。
\end{definition}

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \(f(x)\geq 0\)
    \item \(\int_{-\infty}^{+\infty}f(x)\,\mathrm{d}x=1\)
    \item \(P\{x_1<X\leq x_2\}=F(x_2)-F(x_1)=\int_{x_1}^{x_2}f(x)\,\mathrm{d}x\)
    \item 若\(f(x)\)在点\(x\)处连续，则有\(F'(x)=f(x)\)
\end{enumerate}

\begin{remark}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \(f(x)\Delta x\)在连续型随机变量理论中起的作用与\(P\{X=x_k\}=p_k\)在离散型随机变量理论中所起的作用类似。
    \item \(f(x)\)在某点\(a\)的高度，不反映\(X\)取值的概率，只是意味着\(X\)取\(a\)附近的值的概率的大小。
    \item \(P(A)=0\)，并不意味着\(A=\emptyset\)；\(P(B)=1\)，并不意味着\(B=\Omega\)。
\end{enumerate}
\end{remark}

\textbf{标准正态分布的概率密度}：
\[
\varphi(x)=\frac{1}{\sqrt{2\pi}}e^{-\frac{x^2}{2}},\quad -\infty<x<+\infty
\]

\textbf{标准正态分布的分布函数}：
\[
\Phi(x)=\int_{-\infty}^{x}\frac{1}{\sqrt{2\pi}}e^{-\frac{t^2}{2}}\,\mathrm{d}t,\quad -\infty<x<+\infty
\]

\textbf{标准正态分布的重要公式}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 若\(X\sim N(\mu,\sigma^2)\)，则\(Y=\dfrac{X-\mu}{\sigma}\sim N(0,1)\)
    \item \(P\{c\leq X\leq d\}=\Phi\left(\dfrac{d-\mu}{\sigma}\right)-\Phi\left(\dfrac{c-\mu}{\sigma}\right)\)
    \item \(\Phi(-x)=1-\Phi(x)\)
\end{enumerate}

\section{随机变量的函数的分布}

\subsubsection{离散型随机变量的函数的分布}

如果\(X\)是离散型随机变量，其函数\(Y=g(X)\)也是离散型随机变量。若\(X\)的分布律为

\[
\begin{array}{c|ccccc}
X & x_1 & x_2 & \cdots & x_k & \cdots \\ \hline
p_k & p_1 & p_2 & \cdots & p_k & \cdots
\end{array}
\]

则\(Y=g(X)\)的分布律为

\[
\begin{array}{c|ccccc}
Y=g(X) & g(x_1) & g(x_2) & \cdots & g(x_k) & \cdots \\ \hline
p_k & p_1 & p_2 & \cdots & p_k & \cdots
\end{array}
\]

\subsubsection{连续型随机变量的函数的分布}

如果\(X\)是连续型随机变量，其函数\(Y=g(X)\)也是连续型随机变量。

\textbf{\(Y\)的概率密度计算}：
\begin{itemize}[leftmargin=2em]
    \item 根据\(X\)的密度函数\(f_X(x)\)求出\(Y\)的分布函数
    \[
    F_Y(y)=P\{Y\leq y\}=P\{g(X)\leq y\}=\int_{g(x)\leq y}f_X(x)\,\mathrm{d}x
    \]
    \item 再对\(F_Y(y)\)求导得到\(Y\)的密度函数
\end{itemize}

\textbf{若\(g(X)\)是严格单调函数}，即\(g'(X)<0\)或\(g'(X)>0\)，有
\[
f_Y(y)=\begin{cases}
f_X[h(y)]|h'(y)|, & \alpha<y<\beta, \\[6pt]
0, & \text{其他}
\end{cases}
\]
其中\(\alpha=\min\{g(-\infty),g(+\infty)\}\)，\(\beta=\max\{g(-\infty),g(+\infty)\}\)。

\textbf{若\(g(x)\)在不相叠的区间\(I_1,I_2,\cdots\)上逐段严格单调}，其反函数分别为\(h_1(x),h_2(x),\cdots\)均为连续函数，那么\(Y=g(X)\)是连续型随机变量，其概率密度为
\[
f_Y(y)=f_X[h_1(y)]|h_1'(y)|+f_X[h_2(y)]|h_2'(y)|+\cdots,\quad y\in(\alpha,\beta)
\]

\textbf{直观理解}：\(f_Y(y)\Delta y=f_X[h(y)]\Delta x\)

\begin{example}[题型1：离散型随机变量——求分布律、分布函数、概率、参数]

\textbf{例1} 房间内有10个人，分别佩带1号到10号纪念章，任意选出5个人记录其纪念章的号码，令\(X\)表示其最小号码。

(1) 求\(X\)的分布律。

(2) 求\(P\{X>4\}\)。

\textbf{解}：(1) \(X\)的取值为：\(1,2,3,4,5,6\)，并且
\[
P\{X=1\}=\frac{C_9^4}{C_{10}^5}=\frac{126}{252},\quad
P\{X=2\}=\frac{C_8^4}{C_{10}^5}=\frac{70}{252},
\]
\[
P\{X=3\}=\frac{C_7^4}{C_{10}^5}=\frac{35}{252},\quad
P\{X=4\}=\frac{C_6^4}{C_{10}^5}=\frac{15}{252},
\]
\[
P\{X=5\}=\frac{C_5^4}{C_{10}^5}=\frac{5}{252},\quad
P\{X=6\}=\frac{C_4^4}{C_{10}^5}=\frac{1}{252}.
\]

\(X\)的分布律为

\[
\begin{array}{c|cccccc}
X & 1 & 2 & 3 & 4 & 5 & 6 \\ \hline
P & \dfrac{126}{252} & \dfrac{70}{252} & \dfrac{35}{252} & \dfrac{15}{252} & \dfrac{5}{252} & \dfrac{1}{252}
\end{array}
\]

(2)
\[
P\{X>4\}=P\{X=5\}+P\{X=6\}=\frac{5}{252}+\frac{1}{252}=\frac{6}{252}=0.0238
\]
\end{example}

\begin{example}[例2]
设随机变量\(X\sim B(2,p)\)，且\(P\{X\geq 1\}=\dfrac{5}{9}\)。

(1) 试确定参数\(p\)；

(2) 求\(P\{X=1\}\)。

\textbf{解}：
\[
P\{X=k\}=C_2^k p^k(1-p)^{2-k}\quad(k=0,1,2)
\]

(1) 由\(P\{X\geq 1\}=\dfrac{5}{9}\)，得
\[
P\{X=0\}=1-\frac{5}{9}=\frac{4}{9}=(1-p)^2
\]
解得\(p=\dfrac{1}{3}\)。

(2)
\[
P\{X=1\}=C_2^1\times\frac{1}{3}\times\frac{2}{3}=\frac{4}{9}
\]
\end{example}

\begin{example}[例3]
设离散型随机变量\(X\)的分布函数为
\[
F(x)=\begin{cases}
0, & x<-1, \\[6pt]
a, & -1\leq x<1, \\[6pt]
\dfrac{2}{3}-a, & 1\leq x<2, \\[8pt]
a+b, & x\geq 2
\end{cases}
\]
且\(P\{X=2\}=\dfrac{1}{2}\)，试确定常数\(a,b\)，并求\(X\)的分布。

\textbf{解}：利用分布函数\(F(x)\)的性质\(F(+\infty)=1\)，有\(a+b=1\)。

又
\[
P\{X=2\}=F(2)-F(2-0)=(a+b)-\left(\frac{2}{3}-a\right)=2a+b-\frac{2}{3}=\frac{1}{2}
\]
解得\(a=\dfrac{1}{6}\)，\(b=\dfrac{5}{6}\)。

从而
\[
F(x)=\begin{cases}
0, & x<-1, \\[6pt]
\dfrac{1}{6}, & -1\leq x<1, \\[8pt]
\dfrac{1}{2}, & 1\leq x<2, \\[8pt]
1, & x\geq 2
\end{cases}
\]

\(X\)的分布律为

\[
\begin{array}{c|ccc}
X & -1 & 1 & 2 \\ \hline
P & \dfrac{1}{6} & \dfrac{1}{3} & \dfrac{1}{2}
\end{array}
\]
\end{example}

\begin{example}[题型2：连续型随机变量——求概率密度/分布函数中的参数、已知概率密度求分布函数、已知分布函数求概率密度、概率计算]

\textbf{例4} 设随机变量的密度函数为
\[
f(x)=\begin{cases}
a-bx^2, & -1<x<1, \\[6pt]
0, & \text{其他}
\end{cases}
\]
且概率\(P\{X<\frac{1}{2}\}=\dfrac{27}{32}\)，求常数\(a,b\)的值。

\textbf{解}：由\(\int_{-\infty}^{+\infty}f(x)\,\mathrm{d}x=1\)，有
\[
\int_{-1}^{1}(a-bx^2)\,\mathrm{d}x=1\quad\Longrightarrow\quad 2a-\frac{2b}{3}=1
\]

又
\[
P\left\{X<\frac{1}{2}\right\}=\int_{-1}^{1/2}(a-bx^2)\,\mathrm{d}x=\frac{3a}{2}-\frac{3b}{8}=\frac{27}{32}
\]

联立解得\(a=\dfrac{3}{4}\)，\(b=\dfrac{3}{4}\)。
\end{example}

\begin{example}[例5]
设随机变量\(X\)的分布函数为
\[
F(x)=\begin{cases}
A+Be^{-x}, & x>0, \\[6pt]
0, & x\leq 0
\end{cases}
\]

求：(1) \(A,B\)的值；(2) \(X\)的概率密度；(3) \(P\{X>10\mid X>3\}\)。

\textbf{解}：(1) 由\(F(+\infty)=1\)，得\(A=1\)；由右连续性\(F(0+0)=F(0)\)，得\(A+B=0\)，故\(B=-1\)。

(2) 由(1)知
\[
F(x)=\begin{cases}
1-e^{-x}, & x>0, \\[6pt]
0, & x\leq 0
\end{cases}
\]
于是
\[
f(x)=F'(x)=\begin{cases}
e^{-x}, & x>0, \\[6pt]
0, & x\leq 0
\end{cases}
\]

(3)
\[
P\{X>10\mid X>3\}=\frac{P\{X>10\}}{P\{X>3\}}=\frac{1-F(10)}{1-F(3)}=\frac{e^{-10}}{e^{-3}}=e^{-7}
\]
\end{example}

\begin{example}[题型3：随机变量的函数的分布——求\(Y=g(X)\)的概率密度]

\textbf{通用步骤}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 确定\(Y\)的取值范围
    \item 写出取值范围内\(Y\)的分布函数\(F_Y(y)=P\{Y\leq y\}\)
    \item 对\(F_Y(y)\)求导得到\(f_Y(y)\)
    \item 写出\(f_Y(y)\)在\((-\infty,+\infty)\)上的完整表达式，检查边界处"="的取法
\end{enumerate}

\textbf{公式}：
\begin{itemize}[leftmargin=2em]
    \item 若\(g(X)\)单调：
    \[
    f_Y(y)=\begin{cases}
    f_X[h(y)]|h'(y)|, & \alpha<y<\beta, \\[6pt]
    0, & \text{其他}
    \end{cases}
    \]
    \item 若\(g(X)\)逐段单调：
    \[
    f_Y(y)=f_X[h_1(y)]|h_1'(y)|+f_X[h_2(y)]|h_2'(y)|+\cdots
    \]
\end{itemize}
\end{example}

\begin{example}[例6]
若\(X\sim N(1,2)\)，设\(Y=2X-1\)，求\(P\{Y>1\}\)。

\textbf{解法1}：\(Y=2X-1\sim N(1,8)\)，
\[
P\{Y>1\}=1-P\{Y\leq 1\}=1-P\left\{\frac{Y-1}{\sqrt{8}}\leq\frac{1-1}{\sqrt{8}}\right\}=1-\Phi(0)=0.5
\]

\textbf{解法2}：
\[
P\{Y>1\}=P\{2X-1>1\}=P\{X>1\}=1-P\{X\leq 1\}=0.5
\]
\end{example}

\begin{example}[例7/8]
设随机变量\(X\)服从参数为\(\lambda=2\)的指数分布，试求\(Y=1-e^{-2X}\)的概率密度。

\textbf{解}：随机变量\(X\)的概率密度函数
\[
f(x)=\begin{cases}
2e^{-2x}, & x>0, \\[6pt]
0, & x\leq 0
\end{cases}
\]

\textbf{方法1（分布函数法）}：
\begin{itemize}[leftmargin=2em]
    \item \(Y\)的取值范围：\(X\)在\((0,+\infty)\)内取值时，\(Y\)在\((0,1)\)内取值
    \item \(Y\)的分布函数，在\(0<y<1\)时
    \[
    F_Y(y)=P\{Y\leq y\}=P\{1-e^{-2X}\leq y\}=P\left\{X\leq-\frac{1}{2}\ln(1-y)\right\}=y
    \]
    \item 求导得\(Y\)的概率密度
    \[
    f_Y(y)=F_Y'(y)=\begin{cases}
    1, & y\in(0,1), \\[6pt]
    0, & y\notin(0,1)
    \end{cases}
    \]
\end{itemize}

\textbf{方法2（公式法）}：
\(y=1-e^{-2x}\)单调增加，反函数为\(x=-\dfrac{1}{2}\ln(1-y)\)。

当\(x>0\)时，\(y=1-e^{-2x}\in(0,1)\)。

则有
\[
f_Y(y)=f_X[h(y)]|h'(y)|=2e^{-2\left[-\frac{1}{2}\ln(1-y)\right]}\cdot\frac{1}{2(1-y)}=1
\]

即
\[
f_Y(y)=\begin{cases}
1, & y\in(0,1), \\[6pt]
0, & y\notin(0,1)
\end{cases}
\]
\end{example}

\begin{example}[题型4：经典的连续型随机变量——概率计算、记住典型分布的分布函数/密度函数、灵活运用分布的特点]

\textbf{例9} 设随机变量\(X\)在区间\((-3,6)\)上服从均匀分布，求方程\(x^2+Xx+1=0\)有实根的概率。

\textbf{解}：\(X\)的概率密度函数
\[
f(x)=\begin{cases}
\dfrac{1}{9}, & -3<x<6, \\[10pt]
0, & \text{其他}
\end{cases}
\]

方程\(x^2+Xx+1=0\)有实根的条件是\(\Delta=X^2-4\geq 0\)，即\(|X|\geq 2\)。

\[
P\{\text{方程有实根}\}=P\{X^2-4\geq 0\}=P\{X\leq -2\}+P\{X\geq 2\}
\]
\[
=\int_{-3}^{-2}\frac{1}{9}\,\mathrm{d}x+\int_{2}^{6}\frac{1}{9}\,\mathrm{d}x=\frac{1}{9}+\frac{4}{9}=\frac{5}{9}
\]
\end{example}

\begin{example}[例10]
设有两种鸡蛋混放在一起，其中甲种鸡蛋单只的重量（单位：克）服从\(N(50,25)\)分布，乙种鸡蛋单只的重量（单位：克）服从\(N(45,16)\)分布。设甲种鸡蛋占总只数的\(70\%\)，今从该批鸡蛋中任选一只，

(1) 试求其重量超过55克的概率；

(2) 若已知所抽出的鸡蛋超过55克，问它是甲种蛋的概率是多少？

（已知\(\Phi(1)=0.8413\)，\(\Phi(2.5)=0.9938\)）

\textbf{解}：设\(B=\{\text{选出的鸡蛋是甲种鸡蛋}\}\)，\(\bar{B}=\{\text{选出的鸡蛋是乙种鸡蛋}\}\)，\(A=\{\text{选出的鸡蛋重量超过55克}\}\)。

\(X\)：甲种鸡蛋单只的重量，\(Y\)：乙种鸡蛋单只的重量。

则\(P(B)=0.7\)，\(P(\bar{B})=0.3\)。

\[
P(A\mid B)=P\{X>55\}=1-P\{X\leq 55\}=1-\Phi\left(\frac{55-50}{5}\right)=1-\Phi(1)=1-0.8413=0.1587
\]

\[
P(A\mid\bar{B})=P\{Y>55\}=1-P\{Y\leq 55\}=1-\Phi\left(\frac{55-45}{4}\right)=1-\Phi(2.5)=1-0.9938=0.0062
\]

(1) 由全概率公式：
\[
P(A)=P(B)P(A\mid B)+P(\bar{B})P(A\mid\bar{B})=0.7\times 0.1587+0.3\times 0.0062=0.1129
\]

(2) 由Bayes公式：
\[
P(B\mid A)=\frac{P(A\mid B)P(B)}{P(A)}=\frac{0.1587\times 0.7}{0.1129}=\frac{0.11109}{0.11295}\approx 0.9835
\]
\end{example}

\section{二维随机变量}

\begin{definition}[二维随机变量]
设\(E\)是一个随机试验，它的样本空间是\(S=\{e\}\)，设\(X=X(e)\)和\(Y=Y(e)\)是定义在\(S\)上的随机变量。由它们构成的一个向量\((X,Y)\)，叫做二维随机向量，或\textbf{二维随机变量}。
\end{definition}

\begin{figure}[htbp]

\begin{definition}[二维随机变量的分布函数]
设\((X,Y)\)是一个二维随机变量，则对于任意一对实数\(x,y\)，二元函数
\[
F(x,y)=P\{X\leq x,Y\leq y\}
\]
称为二维随机变量\((X,Y)\)的\textbf{分布函数}，或称为随机变量\(X\)和\(Y\)的\textbf{联合分布函数}。
\end{definition}

\textbf{几何意义}：\(F(x,y)\)可看成是平面上的随机点\((X,Y)\)落在以点\((x,y)\)为右上顶点的左下方的无穷矩形域内的概率。

    \label{fig:joint_dist_region}
\end{figure}

\textbf{重要公式}：

设\(x_1<x_2\)，\(y_1<y_2\)，则
\[
P\{x_1<X\leq x_2,y_1<Y\leq y_2\}=F(x_2,y_2)-F(x_2,y_1)-F(x_1,y_2)+F(x_1,y_1)
\]

\begin{definition}[\(n\)维随机变量]
设\(E\)是一个随机试验，它的样本空间是\(S=\{e\}\)，设\(X_1=X_1(e),X_2=X_2(e),\cdots,X_n=X_n(e)\)是定义在\(S\)上的随机变量，由它们构成的一个\(n\)维向量\((X_1,X_2,\cdots,X_n)\)叫做\(n\)维随机向量或\textbf{\(n\)维随机变量}。

对于任意\(n\)个实数\(x_1,x_2,\cdots,x_n\)，\(n\)元函数
\[
F(x_1,x_2,\cdots,x_n)=P\{X_1\leq x_1,X_2\leq x_2,\cdots,X_n\leq x_n\}
\]
称为随机变量\(X_1,X_2,\cdots,X_n\)的\textbf{联合分布函数}。
\end{definition}

\section{二维离散型随机变量的分布律}

若二维随机变量\((X,Y)\)的取值是有限个或可列无穷个，则称\((X,Y)\)为\textbf{二维离散型随机变量}。

设\((X,Y)\)为二维离散型随机变量，\(X\)的取值为\(x_1,x_2,\cdots,x_i,\cdots\)，\(Y\)的取值为\(y_1,y_2,\cdots,y_j,\cdots\)，则称
\[
p_{ij}=P\{X=x_i,Y=y_j\},\quad(i,j=1,2,\cdots)
\]
为二维离散型随机变量\((X,Y)\)的\textbf{联合分布律}。

离散型随机变量\((X,Y)\)的分布函数为
\[
F(x,y)=\sum_{x_i\leq x}\sum_{y_j\leq y}p_{ij}
\]
对一切满足\(x_i\leq x\)，\(y_j\leq y\)的\(i,j\)项求和。

\section{二维连续型随机变量的概率密度}

\begin{definition}
对于二维随机变量\((X,Y)\)的分布函数\(F(x,y)\)，如果存在非负实函数\(f(x,y)\)，使得对于任意的实数\(x,y\)有
\[
F(x,y)=\int_{-\infty}^{y}\int_{-\infty}^{x}f(u,v)\,\mathrm{d}u\,\mathrm{d}v
\]
则称\((X,Y)\)是\textbf{连续型的二维随机变量}。函数\(f(x,y)\)称为二维随机变量\((X,Y)\)的\textbf{概率密度}，或称为\(X\)和\(Y\)的\textbf{联合概率密度}。
\end{definition}

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{非负性}：\(f(x,y)\geq 0\)
    \item \textbf{规范性}：\(\displaystyle\int_{-\infty}^{+\infty}\int_{-\infty}^{+\infty}f(x,y)\,\mathrm{d}x\,\mathrm{d}y=1\)
    \item 若\(f(x,y)\)在点\((x,y)\)连续，则有
    \[
    f(x,y)=\frac{\partial^2 F(x,y)}{\partial x\,\partial y}
    \]
    \item 设\(G\)是平面上的一个区域，点\((X,Y)\)落在\(G\)内的概率为：
    \[
    P\{(X,Y)\in G\}=\iint_G f(x,y)\,\mathrm{d}x\,\mathrm{d}y
    \]
    应用：估计积分。
\end{enumerate}

\textbf{几何意义}：
\begin{itemize}[leftmargin=2em]
    \item \(z=f(x,y)\)表示空间的一个曲面，
    \[
    \int_{-\infty}^{+\infty}\int_{-\infty}^{+\infty}f(x,y)\,\mathrm{d}x\,\mathrm{d}y=1
    \]
    表示介于曲面\(f(x,y)\)和\(xOy\)平面之间空间区域的体积等于1。
    % 图片：三维曲面图，曲面在xOy平面上方，总体积为1
    \item \(\displaystyle P\{(X,Y)\in G\}=\iint_G f(x,y)\,\mathrm{d}x\,\mathrm{d}y\)的值等于以\(G\)为底，曲面\(z=f(x,y)\)为顶面的柱体体积。
    % 图片：三维柱体示意图，红色边框表示区域G，蓝色柱体表示体积
\end{itemize}

\subsubsection{常用分布}

\textbf{二维均匀分布}：设\(D\)是平面上的有界区域，其面积为\(A\)。如果二维随机变量\((X,Y)\)的密度函数为
\[
f(x,y)=\begin{cases}
\dfrac{1}{A}, & (x,y)\in D, \\[8pt]
0, & (x,y)\notin D
\end{cases}
\]
则称二维随机变量\((X,Y)\)服从区域\(D\)上的\textbf{均匀分布}。

\textbf{二维正态分布}：设二维随机变量\((X,Y)\)的密度函数为
\[
f(x,y)=\frac{1}{2\pi\sigma_1\sigma_2\sqrt{1-\rho^2}}\exp\left\{-\frac{1}{2(1-\rho^2)}\left[\frac{(x-\mu_1)^2}{\sigma_1^2}-\frac{2\rho(x-\mu_1)(y-\mu_2)}{\sigma_1\sigma_2}+\frac{(y-\mu_2)^2}{\sigma_2^2}\right]\right\}
\]
则称随机变量\((X,Y)\)服从参数为\((\mu_1,\mu_2,\sigma_1^2,\sigma_2^2,\rho)\)的\textbf{正态分布}，记作
\[
(X,Y)\sim N(\mu_1,\mu_2,\sigma_1^2,\sigma_2^2,\rho)
\]
其中\(-\infty<\mu_i<+\infty\)（\(i=1,2\)），\(\sigma_i>0\)（\(i=1,2\)），\(-1<\rho<1\)。

\section{边缘分布}

\begin{definition}[边缘分布]
如果\((X,Y)\)是一个二维随机变量，则它的分量\(X\)（或\(Y\)）是一维随机变量。因此，分量\(X\)（或\(Y\)）也有分布，我们称\(X\)（或\(Y\)）的分布为二维随机变量\((X,Y)\)关于\(X\)（或\(Y\)）的\textbf{边缘分布}。

边缘分布也称为边沿分布或边际分布。
\end{definition}

\begin{figure}[htbp]
    \centering
    \includegraphics[width=0.5\textwidth]{figures/fig_joint_margin_relation.png}
    \caption{联合分布与边缘分布的关系}
    \label{fig:joint_margin_relation}
\end{figure}

\subsubsection{已知联合分布函数求边缘分布函数}

设二维随机变量\((X,Y)\)的分布函数为\(F(x,y)\)，则分量\(X\)的分布函数为
\[
F_X(x)=P\{X\leq x\}=P\{X\leq x,Y\leq+\infty\}=\lim_{y\to+\infty}F(x,y)=F(x,+\infty)
\]
分量\(Y\)的分布函数为
\[
F_Y(y)=P\{Y\leq y\}=P\{X\leq+\infty,Y\leq y\}=\lim_{x\to+\infty}F(x,y)=F(+\infty,y)
\]

\subsubsection{已知联合分布律求边缘分布律}

\(X\)和\(Y\)的边缘分布律也可以由下表表示：

\[
\begin{array}{c|ccccc|c}
X\backslash Y & y_1 & y_2 & \cdots & y_j & \cdots & p_{i\cdot} \\ \hline
x_1 & p_{11} & p_{12} & \cdots & p_{1j} & \cdots & p_{1\cdot} \\
x_2 & p_{21} & p_{22} & \cdots & p_{2j} & \cdots & p_{2\cdot} \\
\vdots & \vdots & \vdots & \ddots & \vdots & \ddots & \vdots \\
x_i & p_{i1} & p_{i2} & \cdots & p_{ij} & \cdots & p_{i\cdot} \\
\vdots & \vdots & \vdots & \ddots & \vdots & \ddots & \vdots \\ \hline
p_{\cdot j} & p_{\cdot 1} & p_{\cdot 2} & \cdots & p_{\cdot j} & \cdots & 1
\end{array}
\]

其中
\[
p_{i\cdot}=\sum_{j}p_{ij}=P\{X=x_i\},\quad p_{\cdot j}=\sum_{i}p_{ij}=P\{Y=y_j\}
\]

\subsubsection{已知联合密度函数求边缘密度函数}

若\((X,Y)\)的联合密度函数为\(f(x,y)\)，则随机变量\(X\)的边缘密度函数为
\[
f_X(x)=\int_{-\infty}^{+\infty}f(x,y)\,\mathrm{d}y
\]
随机变量\(Y\)的边缘密度函数为
\[
f_Y(y)=\int_{-\infty}^{+\infty}f(x,y)\,\mathrm{d}x
\]

\begin{remark}
二维正态分布\((X,Y)\sim N(\mu_1,\mu_2,\sigma_1^2,\sigma_2^2,\rho)\)的边缘分布：
\[
X\sim N(\mu_1,\sigma_1^2),\quad Y\sim N(\mu_2,\sigma_2^2)
\]
注意：反之不成立，即由边缘分布为正态分布不能推出联合分布为二维正态分布。
\end{remark}

\section{条件分布}

\subsubsection{离散型随机变量的条件分布}

设\((X,Y)\)是二维离散型随机变量。

对于固定的\(j\)，若\(P\{Y=y_j\}>0\)，则称
\[
P\{X=x_i\mid Y=y_j\}=\frac{P\{X=x_i,Y=y_j\}}{P\{Y=y_j\}}=\frac{p_{ij}}{p_{\cdot j}},\quad i=1,2,\cdots
\]
为在\(Y=y_j\)条件下随机变量\(X\)的\textbf{条件分布律}。

对于固定的\(i\)，若\(P\{X=x_i\}>0\)，则称
\[
P\{Y=y_j\mid X=x_i\}=\frac{P\{X=x_i,Y=y_j\}}{P\{X=x_i\}}=\frac{p_{ij}}{p_{i\cdot}},\quad j=1,2,\cdots
\]
为在\(X=x_i\)条件下随机变量\(Y\)的\textbf{条件分布律}。

\subsubsection{连续型随机变量的条件概率密度}

当\(f_Y(y)>0\)时，随机变量\(X\)在\(Y=y\)条件下的条件密度函数为
\[
f_{X\mid Y}(x\mid y)=\frac{f(x,y)}{f_Y(y)}
\]

当\(f_X(x)>0\)时，随机变量\(Y\)在\(X=x\)条件下的条件密度函数为
\[
f_{Y\mid X}(y\mid x)=\frac{f(x,y)}{f_X(x)}
\]

\begin{remark}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item
    \[
    P(X\leq x\mid Y=y)\approx P(X\leq x\mid y<Y\leq y+\epsilon)=\frac{P(X\leq x,y<Y\leq y+\epsilon)}{P(y<Y\leq y+\epsilon)}
    \]
    \item
    \[
    P(X\leq x\mid Y=y)=\int_{-\infty}^{x}f_{X\mid Y}(x\mid y)\,\mathrm{d}x=\int_{-\infty}^{x}\frac{f(x,y)}{f_Y(y)}\,\mathrm{d}x
    \]
    注意与\(P(X\leq x\mid Y\leq y)=\dfrac{P(X\leq x,Y\leq y)}{P(Y\leq y)}=\dfrac{\displaystyle\int_{-\infty}^{x}\int_{-\infty}^{y}f(x,y)\,\mathrm{d}x\,\mathrm{d}y}{\displaystyle\int_{-\infty}^{y}f_Y(y)\,\mathrm{d}y}\)区分。
\end{enumerate}
\end{remark}

\section{相互独立的随机变量}

\begin{definition}[独立分布]
设\((X,Y)\)是二维随机变量，其联合分布函数为\(F(x,y)\)，随机变量\(X\)的分布函数为\(F_X(x)\)，随机变量\(Y\)的分布函数为\(F_Y(y)\)。如果对于任意的\(x,y\)，有
\[
F(x,y)=F_X(x)F_Y(y)
\]
则称\(X,Y\)是\textbf{相互独立的随机变量}。
\end{definition}

\subsubsection{离散型随机变量的独立性}

设\((X,Y)\)是二维离散型随机变量，其联合分布律为\(p_{ij}=P\{X=x_i,Y=y_j\}\)，随机变量\(X\)的分布律为\(p_{i\cdot}=P\{X=x_i\}\)（\(i=1,2,\cdots\)），随机变量\(Y\)的分布律为\(p_{\cdot j}=P\{Y=y_j\}\)（\(j=1,2,\cdots\)）。

如果对于任意的\(i,j\)，满足
\[
p_{ij}=p_{i\cdot}\,p_{\cdot j}
\]
则称\(X,Y\)是相互独立的随机变量。

\subsubsection{连续型随机变量的独立性}

设\((X,Y)\)是二维连续型随机变量，其联合密度函数为\(f(x,y)\)，随机变量\(X\)的边缘密度函数为\(f_X(x)\)，随机变量\(Y\)的边缘密度函数为\(f_Y(y)\)。

如果对于几乎所有的\(x,y\)，满足
\[
f(x,y)=f_X(x)f_Y(y)
\]
则称\(X,Y\)是相互独立的随机变量。

\subsubsection{\(n\)维随机变量的独立性}

设\((X_1,X_2,\cdots,X_n)\)是\(n\)维随机变量，其联合分布函数为\(F(x_1,x_2,\cdots,x_n)\)，随机变量\(X_i\)的分布函数为\(F_{X_i}(x_i)\)（\(i=1,2,\cdots,n\)）。

如果对于任意的\(n\)维实数组\((x_1,x_2,\cdots,x_n)\)，有
\[
F(x_1,x_2,\cdots,x_n)=F_{X_1}(x_1)F_{X_2}(x_2)\cdots F_{X_n}(x_n)
\]
则\(X_1,X_2,\cdots,X_n\)是相互独立的随机变量。

\section{两个随机变量的函数的分布}

\subsubsection{连续型随机变量函数的分布}

设二维连续型随机变量\((X,Y)\)的联合密度函数为\(f(x,y)\)。

\textbf{1. 如果\(g(x,y)\)是二元连续可微函数}，则\(Z=g(X,Y)\)仍是连续型随机变量，\(Z\)的分布函数为
\[
F_Z(z)=P\{g(X,Y)\leq z\}=\iint_{g(x,y)\leq z}f(x,y)\,\mathrm{d}x\,\mathrm{d}y
\]
\(Z\)的概率密度函数为
\[
f_Z(z)=\frac{\mathrm{d}F_Z(z)}{\mathrm{d}z}
\]

\textbf{2. 如果\(g(x,y)\)的值域是有限或可列集\(\{z_1,z_2,\ldots\}\)}，则\(Z=g(X,Y)\)是离散型随机变量，\(Z\)的分布律为
\[
P\{Z=z_i\}=P\{g(X,Y)=z_i\}=\iint_{g(x,y)=z_i}f(x,y)\,\mathrm{d}x\,\mathrm{d}y,\quad i=1,2,\ldots
\]

\subsubsection{连续型随机变量和的分布}

\textbf{卷积公式}：

设二维连续型随机变量\((X,Y)\)的联合密度函数为\(f(x,y)\)，则\(Z=X+Y\)的密度函数为
\[
f_Z(z)=\int_{-\infty}^{+\infty}f(x,z-x)\,\mathrm{d}x
\]
或
\[
f_Z(z)=\int_{-\infty}^{+\infty}f(z-y,y)\,\mathrm{d}y
\]

如果随机变量\(X\)和\(Y\)相互独立，即\(f(x,y)=f_X(x)f_Y(y)\)，此时\(Z=X+Y\)的密度函数为
\[
f_Z(z)=\int_{-\infty}^{+\infty}f_X(z-y)f_Y(y)\,\mathrm{d}y
\]
或
\[
f_Z(z)=\int_{-\infty}^{+\infty}f_X(x)f_Y(z-x)\,\mathrm{d}x
\]

\textbf{相互独立的正态分布随机变量的可加性}：

如果随机变量\(X\)和\(Y\)相互独立，且\(X\sim N(\mu_1,\sigma_1^2)\)，\(Y\sim N(\mu_2,\sigma_2^2)\)，令\(Z=X+Y\)，则
\[
Z\sim N(\mu_1+\mu_2,\sigma_1^2+\sigma_2^2)
\]

如果随机变量\(X_1,X_2,\cdots,X_n\)相互独立，且\(X_i\sim N(\mu_i,\sigma_i^2)\)（\(i=1,2,\cdots,n\)），\(a_1,a_2,\cdots,a_n\)为\(n\)个实常数，则
\[
Z=\sum_{i=1}^{n}a_iX_i\sim N\left(\sum_{i=1}^{n}a_i\mu_i,\sum_{i=1}^{n}a_i^2\sigma_i^2\right)
\]

\textbf{相互独立的二项分布随机变量的可加性}：

如果随机变量\(X\)和\(Y\)相互独立，且\(X\sim B(m,p)\)，\(Y\sim B(n,p)\)，令\(Z=X+Y\)，则
\[
Z\sim B(m+n,p)
\]

\textbf{相互独立的泊松分布随机变量的可加性}：

如果随机变量\(X\)和\(Y\)相互独立，且\(X\sim P(\lambda_1)\)，\(Y\sim P(\lambda_2)\)，令\(Z=X+Y\)，则
\[
Z\sim P(\lambda_1+\lambda_2)
\]

\subsubsection{连续型随机变量函数的极值分布（最大值、最小值的分布）}

设\(X_1,X_2,\cdots,X_n\)是独立的连续型随机变量，\(X_i\)的分布函数为\(F_i(x)\)（\(i=1,2,\cdots,n\)）。

令\(Z_1=\min(X_1,X_2,\ldots,X_n)\)，则\(Z_1\)的分布函数为
\[
F_{Z_1}(z)=P\{Z_1\leq z\}=1-P\{Z_1>z\}=1-[1-F_1(z)][1-F_2(z)]\cdots[1-F_n(z)]
\]

令\(Z_2=\max(X_1,X_2,\ldots,X_n)\)，则\(Z_2\)的分布函数为
\[
F_{Z_2}(z)=P\{Z_2\leq z\}=F_1(z)F_2(z)\cdots F_n(z)
\]

\subsubsection{一般步骤（分布函数法）}

\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 先求随机变量函数\(Z=g(X,Y)\)的分布函数
    \[
    F_Z(z)=P\{Z\leq z\}=P\{g(X,Y)\leq z\}=\iint_{g(x,y)\leq z}f(x,y)\,\mathrm{d}x\,\mathrm{d}y
    \]
    \item 再求随机变量函数\(Z=g(X,Y)\)的密度函数\(f_Z(z)=F_Z'(z)\)
\end{enumerate}

\textbf{难点}：标明\(z\)的定义域；确定积分区间。

\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 在\(x\)-\(y\)坐标系中画出\(x,y\)的取值范围：\(\{f(x,y)\neq 0\text{ 时的 }x,y\text{ 取值范围}\}\cap\{g(x,y)\leq z\}\)（\(z\)当常数用）
    \item 对\(z\)的取值进行讨论，确定\(x,y\)的积分范围
    \item 写出\(F_Z(z)\)的积分计算式，即\(f(x,y)\)沿\(x\)和\(y\)进行积分
    \item 整理\(F_Z(z)\)表达式，写明\(z\)的取值范围
\end{enumerate}

\textbf{几种常见函数的分布（公式法）}：
\[
\begin{aligned}
Z=X+Y &\quad\Rightarrow\quad f_Z(z)=\int_{-\infty}^{+\infty}f(x,z-x)\,\mathrm{d}x \\
Z=Y/X &\quad\Rightarrow\quad f_Z(z)=\int_{-\infty}^{+\infty}|x|f(x,xz)\,\mathrm{d}x \\
Z=XY &\quad\Rightarrow\quad f_Z(z)=\int_{-\infty}^{+\infty}\frac{1}{|x|}f\left(x,\frac{z}{x}\right)\,\mathrm{d}x
\end{aligned}
\]

\textbf{公式法步骤}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 选定内变量（例如\(Y\)），写出\(f_Z(z)\neq 0\)时\(f_Z(z)\)的表达式。表达式中不再有\(Y\)，仅有\(x\)和\(z\)，并需要对\(x\)积分
    \item 确定\(f_Z(z)\neq 0\)时，\(Z\)和\(X\)的取值范围
    \item 在\(z\)-\(x\)坐标系下，画出\(Z\)和\(X\)的取值范围
    \item 根据3中的图，确定\(x\)积分范围，沿\(x\)方向对\(f(x,z)\)积分
    \item 写出最后结果，标明\(z\)的定义域
\end{enumerate}

\begin{remark}[易错点]
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{区分分布函数、分布律、概率密度}
    \begin{itemize}
        \item 分布函数：\(F(x,y)=P\{X\leq x,Y\leq y\}\)
        \item 分布律（离散型）：\(p_{ij}=P\{X=x_i,Y=y_j\}\)
        \item 概率密度函数（连续型）：\(f(x,y)\)
    \end{itemize}
    \item \textbf{概率密度函数的定义域}
    \begin{itemize}
        \item 不漏写定义域：任何一个概率密度函数，都需标明变量的定义域。
        \item 求谁的密度，写谁的定义域
    \end{itemize}
\end{enumerate}
\end{remark}

\chapter{随机变量的数字特征}

\section{数学期望}

\subsection{一维随机变量的数学期望}

\begin{definition}[离散型随机变量的数学期望]
设离散型随机变量$X$的分布律为
\[
P\{X=x_k\}=p_k,\quad k=1,2,\cdots
\]
若级数$\sum_{k=1}^{\infty}x_k p_k$绝对收敛，即$\sum_{k=1}^{\infty}|x_k|p_k<+\infty$，则称$\sum_{k=1}^{\infty}x_k p_k$为离散型随机变量$X$的\textbf{数学期望}，记为$E(X)$，即
\[
E(X)=\sum_{k=1}^{\infty}x_k p_k.
\]
\end{definition}

\begin{definition}[连续型随机变量的数学期望]
设连续型随机变量$X$的概率密度为$f(x)$，若积分$\int_{-\infty}^{+\infty}|x|f(x)\,\mathrm{d}x$绝对收敛，则称积分$\int_{-\infty}^{+\infty}x f(x)\,\mathrm{d}x$的值为连续型随机变量$X$的\textbf{数学期望}，记为$E(X)$，即
\[
E(X)=\int_{-\infty}^{+\infty}x f(x)\,\mathrm{d}x.
\]
\end{definition}

\subsection{随机变量函数的数学期望}

\begin{definition}[离散型随机变量函数的期望]
若$Y=g(X)$，且$P\{X=x_k\}=p_k,\ k=1,2,\cdots$，若级数$\sum_{k=1}^{\infty}g(x_k)p_k$绝对收敛，则$Y$的数学期望为
\[
E(g(X))=\sum_{k=1}^{\infty}g(x_k)p_k.
\]
\end{definition}

\begin{definition}[连续型随机变量函数的期望]
设连续型随机变量$X$的概率密度为$f(x)$，若$Y=g(X)$，且$\int_{-\infty}^{+\infty}|g(x)|f(x)\,\mathrm{d}x$绝对收敛，则$Y$的数学期望为
\[
E(g(X))=\int_{-\infty}^{+\infty}g(x)f(x)\,\mathrm{d}x.
\]
\end{definition}

\subsection{二维随机变量的数学期望}

设$(X,Y)$为二维随机变量，若$E(X),E(Y)$都存在，则$X$与$Y$的期望分别定义为

\textbf{离散型}：
\[
E(X)=\sum_{i}\sum_{j}x_i p_{ij},\qquad E(Y)=\sum_{i}\sum_{j}y_j p_{ij},
\]
其中$(X,Y)$的联合分布律为$p_{ij}$。

\textbf{连续型}：
\[
E(X)=\int_{-\infty}^{+\infty}\int_{-\infty}^{+\infty}x f(x,y)\,\mathrm{d}x\mathrm{d}y,\qquad
E(Y)=\int_{-\infty}^{+\infty}\int_{-\infty}^{+\infty}y f(x,y)\,\mathrm{d}x\mathrm{d}y,
\]
其中$(X,Y)$的联合概率密度为$f(x,y)$。

\begin{definition}[二维随机变量函数的期望]
若$X,Y$为离散型随机变量，$g(x,y)$是二元函数，$(X,Y)$的联合分布律为$p_{ij}$，且$E[g(X,Y)]$存在，则
\[
E[g(X,Y)]=\sum_{i}\sum_{j}g(x_i,y_j)p_{ij}.
\]

若$X,Y$为连续型随机变量，$g(x,y)$是二元函数，$(X,Y)$的联合密度为$f(x,y)$，且$E[g(X,Y)]$存在，则
\[
E[g(X,Y)]=\int_{-\infty}^{+\infty}\int_{-\infty}^{+\infty}g(x,y)f(x,y)\,\mathrm{d}x\mathrm{d}y.
\]
\end{definition}

\subsection{数学期望的性质}

\begin{property}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 设$C$是常数，则$E(C)=C$。
    \item 设$X$是随机变量，$C$是常数，则$E(CX)=C\,E(X)$。
    \item 设$X,Y$是两个随机变量，则有
    \[
    E(X+Y)=E(X)+E(Y).
    \]
    更一般地，$E(aX+bY+c)=aE(X)+bE(Y)+c$。\\
    \textbf{注意}：加法不需要独立性。
    \item 设$X,Y$是相互独立的随机变量，则有
    \[
    E(XY)=E(X)E(Y).
    \]
    \textbf{注意}：乘法需要独立性。
\end{enumerate}
\end{property}

\section{方差}

\subsection{方差的定义与计算}

\begin{definition}[方差]
设$X$是一个随机变量，若$E\{[X-E(X)]^2\}$存在，则称$E\{[X-E(X)]^2\}$是$X$的\textbf{方差}，记作$D(X)$或$\mathrm{Var}(X)$，即
\[
D(X)=\mathrm{Var}(X)=E\big\{[X-E(X)]^2\big\}.
\]
称$\sqrt{D(X)}$为\textbf{标准差}或\textbf{均方差}，记作$\sigma(X)$。
\end{definition}

\textbf{方差的计算}：
\begin{itemize}[leftmargin=2em]
    \item 离散型随机变量：
    \[
    D(X)=\sum_{k=1}^{\infty}[x_k-E(X)]^2 p_k
    \]
    \item 连续型随机变量：
    \[
    D(X)=\int_{-\infty}^{+\infty}[x-E(X)]^2 f(x)\,\mathrm{d}x
    \]
    \item 常用计算公式：
    \[
    D(X)=E(X^2)-[E(X)]^2
    \]
\end{itemize}

\subsection{方差的性质}

\begin{property}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 设$C$是常数，则$D(C)=0$。
    \item 设$X$是随机变量，$C$是常数，则
    \[
    D(CX)=C^2 D(X).
    \]
    \item 设$X,Y$相互独立，$D(X),D(Y)$存在，则
    \[
    D(X\pm Y)=D(X)+D(Y).
    \]
    \textbf{注意}：方差的可加性要求独立性。\\
    若$X,Y$不独立，则
    \[
    D(X\pm Y)=D(X)+D(Y)\pm 2\big(E(XY)-E(X)E(Y)\big).
    \]
    \item $D(X)=0$的充要条件是$X$以概率$1$取常数$C$，即
    \[
    D(X)=0\iff P\{X=C\}=1.
    \]
\end{enumerate}
\end{property}

\subsection{切比雪夫不等式}

\begin{theorem}[切比雪夫(Chebyshev)不等式]
设随机变量$X$的数学期望$E(X)=\mu$存在，方差$D(X)=\sigma^2$，则对于任意正数$\varepsilon$，有
\[
P\{|X-\mu|\geq\varepsilon\}\leq\frac{\sigma^2}{\varepsilon^2}
\quad\Longleftrightarrow\quad
P\{|X-\mu|<\varepsilon\}\geq 1-\frac{\sigma^2}{\varepsilon^2}.
\]
\end{theorem}

\begin{remark}
切比雪夫不等式给出了在随机变量分布未知的情况下，事件$\{|X-\mu|<\varepsilon\}$概率的下限估计。
\end{remark}

\textbf{回顾：正态分布的$3\sigma$准则}

若$X\sim N(\mu,\sigma^2)$，则当$X\sim N(0,1)$时：
\[
\begin{array}{c}
P\{|X|\leq 1\}=2\Phi(1)-1=0.6826\\[4pt]
P\{|X|\leq 2\}=2\Phi(2)-1=0.9544\\[4pt]
P\{|X|\leq 3\}=2\Phi(3)-1=0.9974
\end{array}
\]
$X$的取值几乎全部集中在$[\mu-3\sigma,\mu+3\sigma]$区间内，超出这个范围的可能性不到$0.3\%$。

\section{协方差与相关系数}

\subsection{协方差}

\begin{definition}[协方差]
随机变量$X$与$Y$的\textbf{协方差}定义为
\[
\mathrm{Cov}(X,Y)=E\big\{[X-E(X)][Y-E(Y)]\big\}.
\]
\end{definition}

\textbf{协方差的常用计算方法}：
\[
\mathrm{Cov}(X,Y)=E(XY)-E(X)E(Y).
\]

\textbf{协方差的性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $\mathrm{Cov}(X,Y)=\mathrm{Cov}(Y,X)$。
    \item $\mathrm{Cov}(aX,bY)=ab\,\mathrm{Cov}(X,Y)$，其中$a,b$为常数。
    \item $\mathrm{Cov}(X_1+X_2,Y)=\mathrm{Cov}(X_1,Y)+\mathrm{Cov}(X_2,Y)$。
    \item $D(X\pm Y)=D(X)+D(Y)\pm 2\,\mathrm{Cov}(X,Y)$。
\end{enumerate}

\subsection{相关系数}

\begin{definition}[相关系数]
随机变量$X$与$Y$的\textbf{相关系数}定义为
\[
\rho_{XY}=\frac{\mathrm{Cov}(X,Y)}{\sqrt{D(X)}\cdot\sqrt{D(Y)}}.
\]
\end{definition}

\textbf{相关系数的性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $|\rho_{XY}|\leq 1$。
    \item $|\rho_{XY}|=1$的充要条件是：存在常数$a,b$，使得
    \[
    P\{Y=a+bX\}=1.
    \]
    即$Y$可被$X$线性表出。
    \item 相关系数只能说明两个变量之间是否存在线性关系，不能得出两个变量是否独立。
\end{enumerate}

\begin{remark}
\begin{itemize}[leftmargin=2em]
    \item 独立$\Rightarrow$不相关；不相关$\nRightarrow$独立。
    \item $X,Y$相互独立时，$\mathrm{Cov}(X,Y)=E(XY)-E(X)E(Y)=0$。
    \item \textbf{特例}：若$X,Y$都服从正态分布，则独立与不相关等价。
\end{itemize}
\end{remark}

\subsection{协方差矩阵}

\begin{definition}[协方差矩阵]
设$n$维随机变量$(X_1,X_2,\cdots,X_n)$的二阶混合中心矩
\[
c_{ij}=\mathrm{Cov}(X_i,X_j)=E\big\{[X_i-E(X_i)][X_j-E(X_j)]\big\},\quad i,j=1,2,\cdots,n
\]
都存在，则称矩阵
\[
\Sigma=\begin{pmatrix}
c_{11} & c_{12} & \cdots & c_{1n}\\
c_{21} & c_{22} & \cdots & c_{2n}\\
\vdots & \vdots & \ddots & \vdots\\
c_{n1} & c_{n2} & \cdots & c_{nn}
\end{pmatrix}
\]
为$n$维随机变量$(X_1,X_2,\cdots,X_n)$的\textbf{协方差矩阵}。
\end{definition}

\begin{example}[《概率论及其应用》9\#8]
令$X_1,X_2,\cdots,X_n$为相互独立的随机变量，每一个都具有均匀分布$P\{X_i=k\}=\frac{1}{N},\ k=1,2,\cdots,N$。令$U_n$和$V_n$分别表示$X_1,X_2,\cdots,X_n$中的最小值和最大值。求$U_n$和$V_n$的分布。

\textbf{解}：对于最大值$V_n=\max\{X_1,X_2,\cdots,X_n\}$：
\[
P\{V_n\leq k\}=P\{X_1\leq k,X_2\leq k,\cdots,X_n\leq k\}=\left(\frac{k}{N}\right)^n.
\]
故
\[
P\{V_n=k\}=P\{V_n\leq k\}-P\{V_n\leq k-1\}=\left(\frac{k}{N}\right)^n-\left(\frac{k-1}{N}\right)^n.
\]
且
\[
E(V_n)\approx\frac{n}{n+1}N.
\]

对于最小值$U_n=\min\{X_1,X_2,\cdots,X_n\}$：
\[
P\{U_n\leq k\}=1-P\{U_n>k\}=1-\left(\frac{N-k}{N}\right)^n.
\]
故
\[
P\{U_n=k\}=P\{U_n\leq k\}-P\{U_n\leq k-1\}=\left(\frac{N-k+1}{N}\right)^n-\left(\frac{N-k}{N}\right)^n.
\]
且
\[
E(U_n)\approx\frac{1}{n+1}N.
\]
\end{example}

\begin{example}[《概率论及其应用》9\#19]
一个人共有$n$把钥匙，试开门时所抽取的钥匙是相互独立且随机的。求试验次数的均值与方差。

(a) 如果以前选出来的不能开门的钥匙不除去；

(b) 如果以前选的无效钥匙除去。

\textbf{解(a)}：设$X$表示试验次数，$X$服从几何分布，分布律为
\[
P\{X=k\}=(1-p)^{k-1}p,\quad p=\frac{1}{n},\ k=1,2,\cdots.
\]
其中$q=1-p$。

期望：
\[
E(X)=\sum_{k=1}^{\infty}k q^{k-1}p=p\cdot\frac{1}{(1-q)^2}=\frac{1}{p}=n.
\]

计算$E(X^2)$：令$S=\sum_{k=1}^{\infty}k^2 q^{k-1}$，利用错位相减法可得
\[
(1-q)^2 S=1+2q+2q^2+\cdots=2\left(\frac{1}{1-q}\right)-1=\frac{1+q}{1-q}
\Rightarrow S=\frac{1+q}{(1-q)^3}.
\]
故
\[
E(X^2)=p\cdot\frac{1+q}{p^3}=\frac{1+q}{p^2},
\qquad
D(X)=E(X^2)-[E(X)]^2=\frac{q}{p^2}=n(n-1).
\]

\textbf{解(b)}：$X$的分布律为
\[
P\{X=k\}=\frac{1}{n},\quad k=1,2,\cdots,n.
\]
则
\[
E(X)=\sum_{k=1}^{n}k\cdot\frac{1}{n}=\frac{n+1}{2},
\]
\[
E(X^2)=\sum_{k=1}^{n}\frac{k^2}{n}=\frac{(n+1)(2n+1)}{6},
\]
\[
D(X)=E(X^2)-[E(X)]^2=\frac{(n+1)(n-1)}{12}=\frac{n^2-1}{12}.
\]
\end{example}

\begin{example}[《概率论及其应用》9\#38]
设$X_1,X_2,\cdots,X_n$为相互独立的随机变量，其均值为$m$，方差为$\sigma^2$。令$\bar{X}=\frac{1}{n}\sum_{k=1}^{n}X_k$，证明
\[
\frac{1}{n-1}E\left(\sum_{k=1}^{n}(X_k-\bar{X})^2\right)=\sigma^2.
\]

\textbf{证}：
\[
\begin{aligned}
E\left(\sum_{k=1}^{n}(X_k-\bar{X})^2\right)
&=E\left(\sum_{k=1}^{n}X_k^2-2\bar{X}\sum_{k=1}^{n}X_k+n\bar{X}^2\right)\\
&=\sum_{k=1}^{n}E(X_k^2)-2nE(\bar{X}^2)+nE(\bar{X}^2)\\
&=\sum_{k=1}^{n}E(X_k^2)-nE(\bar{X}^2).
\end{aligned}
\]

由于$E(X_k^2)=D(X_k)+[E(X_k)]^2=\sigma^2+m^2$，且
\[
E(\bar{X}^2)=\frac{1}{n^2}E\left(\sum_{i=1}^{n}X_i^2+\sum_{i\neq j}X_i X_j\right)
=\frac{n(\sigma^2+m^2)}{n^2}+\frac{n(n-1)m^2}{n^2}
=\frac{\sigma^2}{n}+m^2.
\]

代入得
\[
\sum_{k=1}^{n}(\sigma^2+m^2)-n\left(\frac{\sigma^2}{n}+m^2\right)
=n\sigma^2+nm^2-\sigma^2-nm^2
=(n-1)\sigma^2.
\]
故
\[
\frac{1}{n-1}E\left(\sum_{k=1}^{n}(X_k-\bar{X})^2\right)=\sigma^2.
\]
\end{example}

    \item 运用随机变量数字特征之间的关系式；

\begin{example}[例4]
设随机变量$X$的密度函数为
\[
f(x)=\begin{cases}
c(1-x^2)^\alpha, & -1<x<1,\\
0, & \text{其他},
\end{cases}
\quad (\alpha>-1)
\]
求$E(X)$和$D(X)$。

\textbf{解}：由于$f(x)$是偶函数，故
\[
E(X)=\int_{-\infty}^{+\infty}x f(x)\,\mathrm{d}x=0.
\]

方差
\[
D(X)=E(X^2)=\int_{-1}^{1}c x^2(1-x^2)^\alpha\,\mathrm{d}x.
\]

利用分部积分：
\[
\begin{aligned}
D(X)&=-\frac{c}{2(\alpha+1)}x(1-x^2)^{\alpha+1}\Big|_{-1}^{1}
+\frac{c}{2(\alpha+1)}\int_{-1}^{1}(1-x^2)^{\alpha+1}\,\mathrm{d}x\\
&=\frac{1}{2(\alpha+1)}\int_{-1}^{1}c(1-x^2)^{\alpha}(1-x^2)\,\mathrm{d}x\\
&=\frac{1}{2(\alpha+1)}\left[\int_{-1}^{1}c(1-x^2)^\alpha\,\mathrm{d}x-\int_{-1}^{1}c x^2(1-x^2)^\alpha\,\mathrm{d}x\right]\\
&=\frac{1}{2(\alpha+1)}\big[1-D(X)\big].
\end{aligned}
\]

解得
\[
D(X)=\frac{1}{2\alpha+3}.
\]
\end{example}

\begin{example}[例5]
设随机变量$X$的概率密度$f(x)=\dfrac{1}{\pi(1+x^2)}$，求$E[\min(|X|,1)]$。

\textbf{解}：
\[
\begin{aligned}
E[\min(|X|,1)]
&=\int_{-\infty}^{+\infty}\min(|x|,1)f(x)\,\mathrm{d}x\\
&=\int_{|x|<1}|x|f(x)\,\mathrm{d}x+\int_{|x|\geq 1}f(x)\,\mathrm{d}x\\
&=2\int_{0}^{1}\frac{x}{\pi(1+x^2)}\,\mathrm{d}x+2\int_{1}^{+\infty}\frac{1}{\pi(1+x^2)}\,\mathrm{d}x\\
&=\frac{1}{\pi}\ln(1+x^2)\Big|_{0}^{1}+\frac{2}{\pi}\arctan x\Big|_{1}^{+\infty}\\
&=\frac{1}{\pi}\ln 2+\frac{1}{2}.
\end{aligned}
\]
\end{example}

\subsection{求二维随机变量的数字特征}

\begin{example}[例6]
设二维连续型随机变量$(X,Y)$的联合密度函数为
\[
f(x,y)=\begin{cases}
\dfrac{6}{7}\left(x^2+\dfrac{1}{2}xy\right), & 0<x<1,\ 0<y<2,\\[8pt]
0, & \text{其他},
\end{cases}
\]
求$(X,Y)$的协方差矩阵及相关系数。

\textbf{解}：
\[
E(X)=\int_{0}^{1}\int_{0}^{2}\frac{6}{7}x\left(x^2+\frac{1}{2}xy\right)\mathrm{d}y\mathrm{d}x=\frac{5}{7},
\]
\[
E(Y)=\int_{0}^{1}\int_{0}^{2}\frac{6}{7}y\left(x^2+\frac{1}{2}xy\right)\mathrm{d}y\mathrm{d}x=\frac{8}{7}.
\]

\[
E(X^2)=\int_{0}^{1}\int_{0}^{2}\frac{6}{7}x^2\left(x^2+\frac{1}{2}xy\right)\mathrm{d}y\mathrm{d}x=\frac{39}{70},
\quad
D(X)=\frac{39}{70}-\left(\frac{5}{7}\right)^2=\frac{23}{490}.
\]

\[
E(Y^2)=\int_{0}^{1}\int_{0}^{2}\frac{6}{7}y^2\left(x^2+\frac{1}{2}xy\right)\mathrm{d}y\mathrm{d}x=\frac{34}{21},
\quad
D(Y)=\frac{34}{21}-\left(\frac{8}{7}\right)^2=\frac{46}{147}.
\]

\[
E(XY)=\int_{0}^{1}\int_{0}^{2}\frac{6}{7}xy\left(x^2+\frac{1}{2}xy\right)\mathrm{d}y\mathrm{d}x=\frac{17}{21}.
\]

协方差
\[
\mathrm{Cov}(X,Y)=E(XY)-E(X)E(Y)=\frac{17}{21}-\frac{5}{7}\cdot\frac{8}{7}=-\frac{1}{147}.
\]

因此$(X,Y)$的协方差矩阵为
\[
\Sigma=\begin{pmatrix}
D(X) & \mathrm{Cov}(X,Y)\\[4pt]
\mathrm{Cov}(Y,X) & D(Y)
\end{pmatrix}
=\begin{pmatrix}
\dfrac{23}{490} & -\dfrac{1}{147}\\[8pt]
-\dfrac{1}{147} & \dfrac{46}{147}
\end{pmatrix}.
\]

相关系数
\[
\rho_{XY}=\frac{\mathrm{Cov}(X,Y)}{\sqrt{D(X)}\sqrt{D(Y)}}=-\frac{\sqrt{15}}{69}.
\]
\end{example}

\subsection{利用切比雪夫不等式估算概率}

\begin{example}[例7]
在供暖的季节，住房的平均温度为$20$度，标准差为$2$度，试用切比雪夫不等式估计住房温度与平均温度的偏差的绝对值小于$4$度的概率最少是多少？

\textbf{解}：设住房的温度是随机变量$X$，其期望为$\mu=20$，方差为$\sigma^2=4$。根据切比雪夫不等式，令$\varepsilon=4$，有
\[

\]

\[
P\{|X-20|<4\}=1-P\{|X-20|\geq 4\}\geq 1-\frac{1}{4}=\frac{3}{4}.
\]
所以，住房温度与平均温度的偏差的绝对值小于$4$度的概率最少是$75\%$。
\end{example}

\section{拓展：主成分分析}

\subsection{基本思想}

\textbf{主成分分析}（Principal Component Analysis，简称PCA）是一种常用的数据降维方法。

\textbf{核心问题}：
\begin{itemize}[leftmargin=2em]
    \item 给定$d$维观察空间的数据点，将其投影到更低维度的空间，同时保持尽可能多的信息；
    \item 投影后数据可区分性好（即分得越``开''越好）：最大化投影后数据的方差；
    \item 投影后的低维数据能重构高维原始数据：最小化原始数据点与重构数据点之间的均方误差。
\end{itemize}

\textbf{主成分向量}（The Principal Components）：
\begin{itemize}[leftmargin=2em]
    \item 它们以数据分布最集中区域的中心为原点向外发射的一组向量，即低维空间的基向量；
    \item 第$1$个投影方向对应于数据分布方差最大的方向；
    \item 第$1,2,\ldots,n$个投影方向与之前的投影方向相互正交，对应于剩余补空间中的方差最大的方向。
\end{itemize}

\textbf{求解算法：利用协方差矩阵}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 给定$m$个$n$维数据点$\{x_1,\ldots,x_m\}$，其中$x_i\in\mathbb{R}^n$，计算协方差矩阵
    \[
    \Sigma=\frac{1}{m}\sum_{i=1}^{m}(x_i-\bar{x})(x_i-\bar{x})^\top,\quad \bar{x}=\frac{1}{m}\sum_{i=1}^{m}x_i.
    \]
    \item 对$\Sigma$做奇异值分解（或特征值分解），选取最大的$k$个特征值对应的特征向量作为投影向量，将$n$维数据投影至$k$维空间。
\end{enumerate}

\begin{remark}
PCA投影方向向量 = 协方差矩阵$\Sigma$的特征向量。\\
特征值越大$\Rightarrow$携带的信息越多。
\end{remark}

\section{依概率收敛}

\begin{definition}[依概率收敛]
设$Y_1,Y_2,\ldots,Y_n,\ldots$是一个随机变量序列，$a$是一个常数，若对任意正数$\varepsilon$，有
\[
\lim_{n\to\infty}P\{|Y_n-a|<\varepsilon\}=1,
\]
则称序列$\{Y_n\}$依概率收敛于$a$，记为
\[
Y_n\stackrel{P}{\longrightarrow}a.
\]
\end{definition}

\section{大数定律}

\subsection{（弱）大数定律的定义}

\begin{definition}[弱大数定律]
设$\{X_n,n=1,2,\ldots\}$是一个随机变量序列，而且对每个$n$，$E(X_n)$存在。如果对于任意给定的$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\left\{\left|\frac{1}{n}\sum_{k=1}^{n}X_k-\frac{1}{n}\sum_{k=1}^{n}E(X_k)\right|<\varepsilon\right\}=1,
\]
则称随机变量序列$\{X_n\}$服从（弱）大数定律。
\end{definition}

等价形式：
\[
\frac{1}{n}\sum_{k=1}^{n}X_k\stackrel{P}{\longrightarrow}\frac{1}{n}\sum_{k=1}^{n}E(X_k).
\]

\subsection{常见的大数定律}

\begin{center}

\begin{theorem}[切比雪夫(Chebyshev)大数定律]
设随机变量序列$\{X_n\}$满足如下条件：
\begin{enumerate}[label=(\arabic*),leftmargin=2em]
    \item $X_1,X_2,\ldots,X_n,\ldots$两两相互独立；
    \item 对每一个$n$（$n=1,2,\ldots$），$D(X_n)$存在；
    \item 数列$D(X_n)$有界，即存在常数$c$，使得对任意的$n$（$n=1,2,\ldots$），有$D(X_n)\leq c$。
\end{enumerate}
则对任意$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\left\{\left|\frac{1}{n}\sum_{k=1}^{n}X_k-\frac{1}{n}\sum_{k=1}^{n}E(X_k)\right|<\varepsilon\right\}=1.
\]
即随机变量序列$\{X_n\}$服从大数定律。
\end{theorem}

\begin{theorem}[切比雪夫大数定律的特殊形式]
设随机变量$X_1,X_2,\ldots,X_n,\ldots$是独立同分布的随机变量，且$E(X_k)=\mu$，$D(X_k)=\sigma^2$（$k=1,2,\ldots$），做前$n$个随机变量的算术平均
\[
\bar{X}=\frac{1}{n}\sum_{k=1}^{n}X_k,
\]
则对任意$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\{|\bar{X}-\mu|<\varepsilon\}=\lim_{n\to\infty}P\left\{\left|\frac{1}{n}\sum_{k=1}^{n}X_k-\mu\right|<\varepsilon\right\}=1.
\]
\end{theorem}

\begin{theorem}[泊松(Poisson)大数定律]
如果在独立试验序列中，事件$A$在第$n$次试验中出现概率为$p_n$，设$n_A$是前$n$次试验中事件$A$出现的次数，则对任意$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\left\{\left|\frac{n_A}{n}-\frac{p_1+p_2+\cdots+p_n}{n}\right|<\varepsilon\right\}=1.
\]
\end{theorem}

\begin{theorem}[Bernoulli大数定律]
设$f_A$是$n$次独立重复试验中事件$A$发生的次数，$p$是事件$A$在每次试验中发生的概率，则对于任意$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\left\{\left|\frac{f_A}{n}-p\right|<\varepsilon\right\}=1
\quad\text{或}\quad
\lim_{n\to\infty}P\left\{\left|\frac{f_A}{n}-p\right|\geq\varepsilon\right\}=0.
\]
\end{theorem}

\begin{theorem}[Khinchin大数定律]
设随机变量$X_1,X_2,\ldots,X_n,\ldots$是独立同分布的随机变量序列，且具有数学期望$E(X_k)=\mu$（$k=1,2,\ldots$），做前$n$个变量的算术平均，则对任意$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\left\{\left|\frac{1}{n}\sum_{k=1}^{n}X_k-\mu\right|<\varepsilon\right\}=1.
\]
\textbf{注意}：与前面Chebyshev大数定律的特殊情况相比，这里不要求$D(X_n)$存在。
\end{theorem}

\begin{theorem}[Markov大数定律]
设随机变量$\{X_n\}$是随机变量序列，如果
\[
\lim_{n\to\infty}D\left(\frac{1}{n}\sum_{k=1}^{n}X_k\right)=0
\quad\text{（称为Markov条件）},
\]
则对任意$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\left\{\left|\frac{1}{n}\sum_{k=1}^{n}X_k-\frac{1}{n}\sum_{k=1}^{n}E(X_k)\right|<\varepsilon\right\}=1.
\]
\textbf{注意}：不要求随机变量的独立性，因此提供了一种研究非独立随机变量序列服从大数定律的方法。
\end{theorem}

\section{中心极限定理}

\subsection{中心极限定理的定义}

\begin{definition}[中心极限定理]
设$X_1,X_2,\ldots,X_n,\ldots$是独立的随机变量序列，若$E(X_k)$，$D(X_k)$（$k=1,2,\ldots$）都存在，令
\[
Y_n=\frac{\sum_{k=1}^{n}X_k-\sum_{k=1}^{n}E(X_k)}{\sqrt{\sum_{k=1}^{n}D(X_k)}},
\]
若对任意$x\in\mathbb{R}$，有
\[
\lim_{n\to\infty}P\{Y_n\leq x\}=\frac{1}{\sqrt{2\pi}}\int_{-\infty}^{x}e^{-\frac{t^2}{2}}\,\mathrm{d}t=\Phi(x),
\]
则称该随机变量序列服从中心极限定理。
\end{definition}

等价地，若$\{X_n\}$服从中心极限定理，则当$n$足够大时，$Y_n$近似服从标准正态分布。

\subsection{常见的中心极限定理}

\begin{theorem}[独立同分布的中心极限定理（Lindburg-Levy定理）]
设随机变量$X_1,X_2,\ldots,X_n,\ldots$独立同分布，且$E(X_k)=\mu$，$D(X_k)=\sigma^2>0$（$k=1,2,\ldots$），则随机变量之和$\sum_{k=1}^{n}X_k$的标准化变量
\[
Y_n=\frac{\sum_{k=1}^{n}X_k-n\mu}{\sqrt{n}\sigma}
\]
的分布函数$F_n(x)$对于任意的$x$满足
\[
\lim_{n\to\infty}F_n(x)=\lim_{n\to\infty}P\left\{\frac{\sum_{k=1}^{n}X_k-n\mu}{\sqrt{n}\sigma}\leq x\right\}=\frac{1}{\sqrt{2\pi}}\int_{-\infty}^{x}e^{-\frac{t^2}{2}}\,\mathrm{d}t=\Phi(x).
\]
\end{theorem}

\begin{theorem}[李雅普诺夫(Lyapunov)定理]
设随机变量$X_1,X_2,\ldots,X_n,\ldots$相互独立，且$E(X_k)=\mu_k$，$D(X_k)=\sigma_k^2>0$（$k=1,2,\ldots$），记$B_n^2=\sum_{k=1}^{n}\sigma_k^2$。若存在$\delta>0$，使得当$n\to\infty$时
\[
\frac{1}{B_n^{2+\delta}}\sum_{k=1}^{n}E|X_k-\mu_k|^{2+\delta}\to 0,
\]
则随机变量之和$\sum_{k=1}^{n}X_k$的标准化变量
\[
Z_n=\frac{\sum_{k=1}^{n}X_k-\sum_{k=1}^{n}\mu_k}{B_n}
\]
的分布函数$F_n(x)$对于任意的$x$满足
\[
\lim_{n\to\infty}F_n(x)=\lim_{n\to\infty}P\{Z_n\leq x\}=\frac{1}{\sqrt{2\pi}}\int_{-\infty}^{x}e^{-\frac{t^2}{2}}\,\mathrm{d}t=\Phi(x).
\]
\end{theorem}

\begin{theorem}[棣莫佛-拉普拉斯(De Moivre-Laplace)定理]
设随机变量$\eta_n$（$n=1,2,\ldots$）服从参数为$n,p$（$0<p<1$）的二项分布，则对于任意$x$，有
\[
\lim_{n\to\infty}P\left\{\frac{\eta_n-np}{\sqrt{np(1-p)}}\leq x\right\}=\frac{1}{\sqrt{2\pi}}\int_{-\infty}^{x}e^{-\frac{t^2}{2}}\,\mathrm{d}t=\Phi(x).
\]
\end{theorem}

\begin{remark}
De Moivre-Laplace定理可以认为是独立同分布的中心极限定理的一个特例：$\eta_n$可以分解成$n$个相互独立、服从同一（0-1）分布的随机变量$X_1,X_2,\ldots,X_n$之和。
\end{remark}

\subsection{二项分布的逼近}

对于一个参数为$n,p$的二项分布，当$n\to\infty$时：
\begin{enumerate}[label=(\arabic*),leftmargin=2em]
    \item 若$np=\lambda$，则该二项分布逼近参数为$\lambda$的泊松分布（泊松定理）；
    \item 若$np\to\infty$，则该二项分布逼近正态分布（De Moivre-Laplace定理）。
\end{enumerate}

\textbf{在实际计算中}：
\begin{itemize}[leftmargin=2em]
    \item 如果$n$很大，但是$p$很小或$q=1-p$很小，应该用泊松定理去近似二项分布；
    \item 如果$n,np$和$nq$都较大，应该用中心极限定理去近似。
\end{itemize}

\section{典型例题}

\subsection{题型1：证明依概率收敛性}

\begin{example}[例1]
设独立同分布随机变量序列$\{X_n\}$均服从$[0,a]$上的均匀分布，令
\[
Y_n=\max\{X_1,X_2,\ldots,X_n\},
\]
试证明：$Y_n\stackrel{P}{\longrightarrow}a$。

\textbf{思路}：要证$Y_n\stackrel{P}{\longrightarrow}a$，即证对任给的$\varepsilon>0$，有
\[
\lim_{n\to\infty}P\{|Y_n-a|>\varepsilon\}=0.
\]

\textbf{证}：由于$\{X_n\}$均服从$[0,a]$上的均匀分布，因此，对任意$n$，当$\varepsilon>a$时，有$P\{|Y_n-a|>\varepsilon\}=0$。

对任意的$0<\varepsilon<a$，有
\[
\begin{aligned}
P\{|Y_n-a|>\varepsilon\}
&=P\{Y_n<a-\varepsilon\}+P\{Y_n>a+\varepsilon\}\\
&=P\{\max(X_1,X_2,\ldots,X_n)<a-\varepsilon\}\\
&=P\{X_1<a-\varepsilon\}P\{X_2<a-\varepsilon\}\cdots P\{X_n<a-\varepsilon\}\\
&=\left(\frac{a-\varepsilon}{a}\right)^n.
\end{aligned}
\]
从而$\lim_{n\to\infty}P\{|Y_n-a|>\varepsilon\}=0$，即$Y_n\stackrel{P}{\longrightarrow}a$。
\end{example}

\begin{example}[例2]
设$X_1,X_2,\ldots,X_n$是独立同分布的随机变量序列，$E(X_i)=\mu$，$D(X_i)=\sigma^2$均存在，证明
\[
Y_n=\frac{2}{n(n+1)}\sum_{i=1}^{n}iX_i\stackrel{P}{\longrightarrow}\mu.
\]

\textbf{思路}：由依概率收敛的定义，即证$\forall\varepsilon>0$，有$\lim_{n\to\infty}P\{|Y_n-\mu|<\varepsilon\}=1$；考虑到$E(Y_n)=\mu$，$D(Y_n)$存在，可由切比雪夫不等式证出。

\textbf{证}：
\[
E(Y_n)=\frac{2}{n(n+1)}\sum_{i=1}^{n}iE(X_i)=\frac{2\mu}{n(n+1)}\sum_{i=1}^{n}i=\mu.
\]
\[
D(Y_n)=\frac{4}{n^2(n+1)^2}\sum_{i=1}^{n}i^2D(X_i)=\frac{4\sigma^2}{n^2(n+1)^2}\cdot\frac{n(n+1)(2n+1)}{6}=\frac{2(2n+1)\sigma^2}{3n(n+1)}.
\]

随机变量$Y_n$的数学期望$E(Y_n)=\mu$，且方差$D(Y_n)$存在，则对于任意正数$\varepsilon>0$，由切比雪夫不等式得：
\[
P\{|Y_n-\mu|\geq\varepsilon\}\leq\frac{D(Y_n)}{\varepsilon^2}=\frac{2\sigma^2(2n+1)}{3n(n+1)\varepsilon^2}.
\]
故
\[
\lim_{n\to\infty}P\{|Y_n-\mu|\geq\varepsilon\}\leq\lim_{n\to\infty}\frac{2\sigma^2(2n+1)}{3n(n+1)\varepsilon^2}=0,
\]
即$Y_n\stackrel{P}{\longrightarrow}\mu$。
\end{example}

\subsection{题型2：利用中心极限定理估算概率}

\begin{example}[例3]
设每颗炮弹命中目标的概率为$0.01$，求$500$发炮弹中命中$5$发的概率。

\textbf{解}：设$X$表示命中的炮弹数，则$X\sim B(500,0.01)$。

(1) 直接计算：
\[
P\{X=5\}=C_{500}^{5}\times 0.01^5\times 0.99^{495}.
\]

(2) 应用正态逼近，近似地
\[
\frac{X-500\times 0.01}{\sqrt{500\times 0.01\times 0.99}}\sim N(0,1),
\]
\[
P\{X=5\}=P\{4.5<X<5.5\}\approx\Phi\left(\frac{5.5-5}{\sqrt{4.95}}\right)-\Phi\left(\frac{4.5-5}{\sqrt{4.95}}\right).
\]
\end{example}

\begin{example}[例4]
$100$个独立工作（工作的概率为$0.9$）的部件组成一个系统，求系统中至少有$85$个部件工作的概率。

\textbf{解}：令$X_i=1$表示第$i$个部件正常工作，反之记为$X_i=0$，$E(X_i)=0.9$，$D(X_i)=0.09$。记$Y=X_1+X_2+\cdots+X_{100}$，因为$X_i$独立同分布，由中心极限定理
\[
\frac{Y-100E(X_i)}{\sqrt{100D(X_i)}}\stackrel{\text{近似}}{\sim}N(0,1).
\]
由此得
\[
P\{Y\geq 85\}=1-P\{Y<85\}\approx 1-\Phi\left(\frac{85-90}{\sqrt{9}}\right)=1-\Phi(-1.67)=\Phi(1.67)=0.9525.
\]
\end{example}

\begin{example}[例5]
有$200$台独立工作（工作的概率为$0.7$）的机床，每台机床工作时需$15$kw电力。问共需多少电力，才可有$95\%$的可能性保证供电充足？

\textbf{解}：用$X_i=1$表示第$i$台机床正常工作，反之记为$X_i=0$。$E(X_i)=0.7$，$D(X_i)=0.21$。记$Y=X_1+X_2+\cdots+X_{200}$，由中心极限定理
\[
\frac{Y-200E(X_i)}{\sqrt{200D(X_i)}}\stackrel{\text{近似}}{\sim}N(0,1).
\]
设供电量为$y$，供电充足即为$15Y\leq y$，则
\[
P\{15Y\leq y\}\approx\Phi\left(\frac{y/15-140}{\sqrt{42}}\right)\geq 0.95.
\]
解得：$y\geq 2260.4$。
\end{example}

\begin{example}[例6]
假设某大学中报名选修统计课的学生人数服从参数为$100$的泊松分布。负责开课的教授决定，如果学生的报名人数不少于$120$人，就分成两个班授课。试问：教授将讲授两个班的概率是多少？

\textbf{解}：设报名选修统计课的学生人数为$X\sim P(100)$。设$X_1,X_2,\ldots,X_{100}$相互独立且$X_i\sim P(1)$。由泊松分布的可加性知
\[
X=X_1+X_2+\cdots+X_{100}\sim P(100).
\]
又因为$E(X_i)=1$，$D(X_i)=1$，由独立同分布的中心极限定理
\[
P\{X\geq 120\}=P\left\{\frac{\sum_{i=1}^{100}X_i-100}{\sqrt{100}}\geq\frac{120-100}{\sqrt{100}}\right\}\approx 1-\Phi(2)=0.0228.
\]
\end{example}

\begin{example}[例7]
用调查对象中的收看比例$k/n$作为某电视节目的收视率$p$的估计。要有$90\%$的把握，使$k/n$与$p$的差异不大于$0.05$，问至少要调查多少对象？

\textbf{解}：用$X_i=1$表示第$i$个调查对象观看某电视节目，否则$X_i=0$，$E(X_i)=p$，$D(X_i)=p(1-p)$。$Y_n=X_1+\cdots+X_n$表示前$n$个调查对象中观看节目的人数。由中心极限定理
\[
\frac{Y_n-np}{\sqrt{np(1-p)}}\stackrel{\text{近似}}{\sim}N(0,1).
\]
根据题意
\[
P\left\{\left|\frac{Y_n}{n}-p\right|<0.05\right\}\approx 2\Phi\left(0.05\sqrt{\frac{n}{p(1-p)}}\right)-1\geq 0.90.
\]
解得：$n\geq 272.25$，即$n=273$。
\end{example}

\begin{example}[例8]
一生产线生产的产品成箱包装，每箱的重量是随机的。假设每箱平均重$50$千克，标准差为$5$千克。若用最大载重量为$5$吨的汽车承运，试利用中心极限定理说明每辆车最多可以装多少箱，才能保证不超载的概率大于$0.977$。

\textbf{解}：设$X_i$（$i=1,2,\ldots,n$）是装运第$i$箱的重量，$n$是所求箱数。则载重量$Y_n=X_1+X_2+\cdots+X_n$。$\mu=E(X_i)=50$，$\sigma^2=D(X_i)=25$。

根据独立同分布的中心极限定理知$\frac{Y_n-50n}{5\sqrt{n}}$的极限分布是标准正态分布。
\[
P\{\text{不超载}\}=P\{0<Y_n\leq 5000\}\approx\Phi\left(\frac{1000-10n}{\sqrt{n}}\right)-\Phi(-10\sqrt{n})>0.977.
\]
由$\Phi(2)=0.977$，得$\frac{1000-10n}{\sqrt{n}}>2$，解得$n<98.0199$，故最多可装$98$箱。
\end{example}

\begin{example}[例9]
某保险公司的老年人寿保险有$1$万人参加，每人每年交$200$元。若老人在该年内死亡，公司付给家属$1$万元。设老年人死亡率为$0.017$，试求保险公司在一年内的这项保险中亏本的概率。

\textbf{解}：设$X$为一年中投保老人的死亡数，则$X\sim B(10000,0.017)$。由棣莫佛-拉普拉斯定理
\[
P\{10000X>10000\times 200\}=P\{X>200\}\approx 1-\Phi\left(\frac{200-170}{\sqrt{167.1}}\right)\approx 1-\Phi(2.321)\approx 0.01.
\]
\end{example}

\subsection{概率的计算}

\textbf{（1）条件概率}
\[
P(B|A)=\frac{P(AB)}{P(A)},\quad (P(A)>0).
\]
缩小样本空间法（适用于古典概型）：
\[
P(B|A)=\frac{n_{AB}}{n_A}.
\]

\textbf{（2）乘法公式}
\[
P(AB)=P(A)P(B|A),\quad P(A)>0.
\]
\[
P(A_1A_2\cdots A_n)=P(A_1)P(A_2|A_1)P(A_3|A_1A_2)\cdots P(A_n|A_1A_2\cdots A_{n-1}).
\]

\textbf{（3）全概率公式}（已知原因，求结果）
\[
P(A)=\sum_{i=1}^{n}P(B_i)P(A|B_i).
\]

\textbf{（4）Bayes（逆概）公式}（已知结果，求原因）
\[
P(B_k|A)=\frac{P(A|B_k)P(B_k)}{\sum_{i=1}^{n}P(A|B_i)P(B_i)},\quad k=1,2,\ldots,n.
\]

\subsection{随机事件的独立性}

\textbf{（1）两事件独立的定义}
\[
P(AB)=P(A)P(B).
\]

\textbf{（2）两事件独立性的性质}
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 事件$A$与$B$相互独立的充分必要条件为$P(B|A)=P(B)$（$P(A)>0$）。
    \item 若随机事件$A$与$B$相互独立，则$\bar{A}$与$B$、$A$与$\bar{B}$、$\bar{A}$与$\bar{B}$也相互独立。
    \item 必然事件$S$与任意随机事件$A$相互独立；不可能事件$\emptyset$与任意随机事件$A$相互独立。
\end{enumerate}

\textbf{注意1}：两事件相互独立与互不相容的区别
\begin{itemize}[leftmargin=2em]
    \item ``$A$与$B$互不相容''，指两事件不能同时发生，即$AB=\emptyset$。
    \item ``$A$与$B$相互独立''，指$A$是否发生不影响$B$发生的概率。
    \item 若$P(A)P(B)\neq 0$，则互不相容与相互独立不能同时成立。
\end{itemize}

\textbf{（3）三个事件的独立性}

设$A,B,C$是三个随机事件，如果
\[
\begin{cases}
P(AB)=P(A)P(B),\\
P(BC)=P(B)P(C),\\
P(AC)=P(A)P(C),\\
P(ABC)=P(A)P(B)P(C),
\end{cases}
\]
则称$A,B,C$是相互独立的随机事件。

\textbf{注意}：四个等式缺一不可。

\textbf{（4）$n$个事件的独立性}

设$A_1,A_2,\ldots,A_n$为$n$个随机事件，若对任意$2\leq k\leq n$和任意$1\leq i_1<i_2<\cdots<i_k\leq n$，都有
\[
P(A_{i_1}A_{i_2}\cdots A_{i_k})=P(A_{i_1})P(A_{i_2})\cdots P(A_{i_k}),
\]
则称$A_1,A_2,\ldots,A_n$相互独立。

\subsection{Bernoulli概型}

\textbf{（1）Bernoulli试验}：若随机试验$E$只有两个结果（成功与失败）。

\textbf{（2）$n$重Bernoulli试验}：独立重复地进行$n$次Bernoulli试验。

\textbf{（3）$n$重Bernoulli试验中恰好成功$k$次的概率}
\[
P(B_{n,k})=C_n^k p^k(1-p)^{n-k},\quad k=0,1,2,\ldots,n.
\]

\subsection{常用的离散型随机变量}

\textbf{（1）Bernoulli分布（两点分布）}

若$X$的分布律为
\[
P\{X=1\}=p,\quad P\{X=0\}=1-p=q,
\]
则称$X\sim B(1,p)$。

\textbf{（2）二项分布}

若$X$的分布律为
\[
P\{X=k\}=C_n^k p^k(1-p)^{n-k},\quad k=0,1,2,\ldots,n,
\]
则称$X\sim B(n,p)$。

\textbf{（3）Poisson分布}

若$X$的分布律为
\[
P\{X=k\}=\frac{\lambda^k}{k!}e^{-\lambda},\quad k=0,1,2,\ldots,
\]
则称$X\sim P(\lambda)$。

\textbf{Poisson定理}：设$X_n\sim B(n,p_n)$，若$\lim_{n\to\infty}np_n=\lambda>0$，则
\[
\lim_{n\to\infty}P\{X_n=k\}=\frac{\lambda^k}{k!}e^{-\lambda}.
\]

\textbf{（4）几何分布}

若$X$的分布律为
\[
P\{X=k\}=q^{k-1}p,\quad k=1,2,\ldots,
\]
则称$X$服从参数为$p$的几何分布。

\textbf{（5）超几何分布}

若$X$的分布律为
\[
P\{X=k\}=\frac{C_M^k C_{N-M}^{n-k}}{C_N^n},\quad k=0,1,\ldots,\min(M,n),
\]
则称$X$服从参数为$(M,N,n)$的超几何分布。

\subsection{常用的连续型随机变量}

\textbf{（1）均匀分布}：$X\sim U[a,b]$
\[
f(x)=\begin{cases}
\dfrac{1}{b-a}, & a\leq x\leq b,\\[8pt]
0, & \text{其他}.
\end{cases}
\]

\textbf{（2）指数分布}：$X\sim E(\lambda)$
\[
f(x)=\begin{cases}
\lambda e^{-\lambda x}, & x>0,\\[6pt]
0, & x\leq 0.
\end{cases}
\]

\textbf{（3）正态分布}：$X\sim N(\mu,\sigma^2)$
\[
f(x)=\frac{1}{\sqrt{2\pi}\sigma}e^{-\frac{(x-\mu)^2}{2\sigma^2}},\quad -\infty<x<+\infty.
\]

\textbf{标准正态分布}：$X\sim N(0,1)$
\[
\varphi(x)=\frac{1}{\sqrt{2\pi}}e^{-\frac{x^2}{2}},\quad \Phi(x)=\int_{-\infty}^{x}\frac{1}{\sqrt{2\pi}}e^{-\frac{t^2}{2}}\,\mathrm{d}t.
\]

\textbf{重要公式}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 若$X\sim N(\mu,\sigma^2)$，则$Y=\dfrac{X-\mu}{\sigma}\sim N(0,1)$。
    \item $P\{c\leq X\leq d\}=\Phi\left(\dfrac{d-\mu}{\sigma}\right)-\Phi\left(\dfrac{c-\mu}{\sigma}\right)$。
    \item $\Phi(-x)=1-\Phi(x)$。
\end{enumerate}

\textbf{$3\sigma$准则}：
\[
P\{|X|\leq 1\}=0.6826,\quad P\{|X|\leq 2\}=0.9544,\quad P\{|X|\leq 3\}=0.9974.
\]

\subsection{随机变量的函数的分布}

\textbf{离散型}：若$X$的分布律已知，$Y=g(X)$的分布律为
\[
P\{Y=g(x_k)\}=p_k.
\]

\textbf{连续型}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 根据$X$的密度函数$f_X(x)$求出$Y$的分布函数
    \[
    F_Y(y)=P\{Y\leq y\}=\int_{g(x)\leq y}f_X(x)\,\mathrm{d}x.
    \]
    \item 再对$F_Y(y)$求导得到$Y$的密度函数。
\end{enumerate}

\textbf{若$g(X)$严格单调}：
\[
f_Y(y)=\begin{cases}
f_X[h(y)]|h'(y)|, & \alpha<y<\beta,\\[6pt]
0, & \text{其他}.
\end{cases}
\]

\subsection{联合分布、边缘分布与条件分布}

\textbf{离散型}：联合分布律$p_{ij}=P\{X=x_i,Y=y_j\}$。

\textbf{连续型}：联合密度$f(x,y)$满足
\[
F(x,y)=\int_{-\infty}^{y}\int_{-\infty}^{x}f(u,v)\,\mathrm{d}u\mathrm{d}v.
\]

\textbf{边缘密度}：
\[
f_X(x)=\int_{-\infty}^{+\infty}f(x,y)\,\mathrm{d}y,\qquad
f_Y(y)=\int_{-\infty}^{+\infty}f(x,y)\,\mathrm{d}x.
\]

\textbf{条件密度}：
\[
f_{X|Y}(x|y)=\frac{f(x,y)}{f_Y(y)}\quad(f_Y(y)>0),\qquad
f_{Y|X}(y|x)=\frac{f(x,y)}{f_X(x)}\quad(f_X(x)>0).
\]

\textbf{联合、边缘、条件的关系}：
\[
f(x,y)=f_X(x)f_{Y|X}(y|x)=f_Y(y)f_{X|Y}(x|y).
\]

\subsection{随机变量的独立性}

$X$与$Y$相互独立$\iff F(x,y)=F_X(x)F_Y(y)$

$\iff$离散型：$p_{ij}=p_{i\cdot}\,p_{\cdot j}$

$\iff$连续型：$f(x,y)=f_X(x)f_Y(y)$

\subsection{二维随机变量函数的分布}

\textbf{和$Z=X+Y$的分布}：
\[
f_Z(z)=\int_{-\infty}^{+\infty}f(x,z-x)\,\mathrm{d}x
=\int_{-\infty}^{+\infty}f(z-y,y)\,\mathrm{d}y.
\]

若$X,Y$独立（卷积公式）：
\[
f_Z(z)=\int_{-\infty}^{+\infty}f_X(x)f_Y(z-x)\,\mathrm{d}x.
\]

\textbf{相互独立的正态分布随机变量之和}：

若$X\sim N(\mu_1,\sigma_1^2)$，$Y\sim N(\mu_2,\sigma_2^2)$，且$X,Y$独立，则
\[
Z=X+Y\sim N(\mu_1+\mu_2,\sigma_1^2+\sigma_2^2).
\]

更一般地，若$X_i\sim N(\mu_i,\sigma_i^2)$且相互独立，则
\[
\sum_{i=1}^{n}a_iX_i\sim N\left(\sum_{i=1}^{n}a_i\mu_i,\sum_{i=1}^{n}a_i^2\sigma_i^2\right).
\]

\section{第四章 随机变量的数字特征}

\textbf{重要知识点}：
\begin{itemize}[leftmargin=2em]
    \item 数学期望、方差、相关系数的性质和计算
    \item 随机变量函数的数字特征
\end{itemize}

\subsection{数学期望}

\textbf{离散型}：$E(X)=\sum_{k=1}^{\infty}x_kp_k$。

\textbf{连续型}：$E(X)=\int_{-\infty}^{+\infty}xf(x)\,\mathrm{d}x$。

\textbf{随机变量函数的期望}：
\[
E(g(X))=\sum_{k=1}^{\infty}g(x_k)p_k\quad\text{或}\quad
E(g(X))=\int_{-\infty}^{+\infty}g(x)f(x)\,\mathrm{d}x.
\]

\textbf{二维}：
\[
E(X)=\iint xf(x,y)\,\mathrm{d}x\mathrm{d}y,\quad
E(g(X,Y))=\iint g(x,y)f(x,y)\,\mathrm{d}x\mathrm{d}y.
\]

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $E(C)=C$。
    \item $E(CX)=CE(X)$。
    \item $E(X+Y)=E(X)+E(Y)$（加法不需要独立性）。
    \item $E(XY)=E(X)E(Y)$（乘法需要独立性）。
\end{enumerate}

\subsection{方差}

\textbf{定义}：$D(X)=E\{[X-E(X)]^2\}=E(X^2)-[E(X)]^2$。

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $D(C)=0$。
    \item $D(CX)=C^2D(X)$。
    \item 若$X,Y$独立，则$D(X\pm Y)=D(X)+D(Y)$。
    \item $D(X)=0\iff P\{X=C\}=1$。
\end{enumerate}

\textbf{切比雪夫不等式}：
\[
P\{|X-\mu|\geq\varepsilon\}\leq\frac{\sigma^2}{\varepsilon^2}
\quad\Longleftrightarrow\quad
P\{|X-\mu|<\varepsilon\}\geq 1-\frac{\sigma^2}{\varepsilon^2}.
\]

\subsection{协方差与相关系数}

\textbf{协方差}：
\[
\mathrm{Cov}(X,Y)=E\{[X-E(X)][Y-E(Y)]\}=E(XY)-E(X)E(Y).
\]

\textbf{相关系数}：
\[
\rho_{XY}=\frac{\mathrm{Cov}(X,Y)}{\sqrt{D(X)}\sqrt{D(Y)}}.
\]

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $|\rho_{XY}|\leq 1$。
    \item $|\rho_{XY}|=1\iff$存在常数$a,b$，使得$P\{Y=a+bX\}=1$。
    \item 独立$\Rightarrow$不相关；不相关$\nRightarrow$独立。
    \item 若$X,Y$都服从正态分布，则独立与不相关等价。
\end{enumerate}

\section{总体、个体与样本}

\begin{definition}[总体与个体]
\begin{itemize}[leftmargin=2em]
    \item \textbf{总体}：试验的全部可能的观察值。
    \item \textbf{个体}：总体中每一个可能的观察值。
\end{itemize}
\end{definition}

\begin{definition}[简单随机样本]
设$X$是具有分布函数$F$的随机变量，$X_1,X_2,\ldots,X_n$是具有同一分布函数$F$、相互独立的随机变量，则称$X_1,X_2,\ldots,X_n$为从分布函数$F$（或总体$F$、或总体$X$）得到的容量为$n$的\textbf{简单随机样本}，简称\textbf{样本}。
\end{definition}

\section{经验分布函数}

\begin{definition}[经验分布函数]
设$X_1,X_2,\ldots,X_n$是总体$F$的一个样本。用$S(x)$（$-\infty<x<+\infty$）表示$X_1,X_2,\ldots,X_n$中不大于$x$的随机变量个数，定义\textbf{经验分布函数}（样本分布函数）为：
\[
F_n(x)=\frac{1}{n}S(x),\quad -\infty<x<+\infty.
\]
\end{definition}

\textbf{经验分布函数的观察值}：一般地，设$x_1,x_2,\ldots,x_n$是总体$F$的一个样本值，先将$x_1,x_2,\ldots,x_n$按自小到大排序，并重新编号满足：
\[
x_{(1)}\leq x_{(2)}\leq\cdots\leq x_{(n)}.
\]
则经验分布函数$F_n(x)$的观察值为：
\[
F_n(x)=
\begin{cases}
0, & \text{若 }x<x_{(1)},\\[4pt]
\dfrac{k}{n}, & \text{若 }x_{(k)}\leq x<x_{(k+1)},\\[8pt]
1, & \text{若 }x\geq x_{(n)}.
\end{cases}
\]

\section{统计量}

\begin{definition}[统计量]
完全由样本所决定的量，叫做\textbf{统计量}。
\end{definition}

设$X_1,X_2,\ldots,X_n$是来自总体$X$的一个样本，常用的样本数字特征统计量如下：

\textbf{样本均值}：
\[
\bar{X}=\frac{1}{n}\sum_{i=1}^{n}X_i
\quad\text{（反映了总体均值的信息）}
\]

\textbf{样本方差}：
\[
S^2=\frac{1}{n-1}\sum_{i=1}^{n}(X_i-\bar{X})^2
=\frac{1}{n-1}\left(\sum_{i=1}^{n}X_i^2-n\bar{X}^2\right)
\quad\text{（反映了总体方差的信息）}
\]

\textbf{样本标准差}：
\[
S=\sqrt{S^2}
\quad\text{（反映了总体标准差的信息）}
\]

\textbf{$k$阶样本原点矩}：
\[
A_k=\frac{1}{n}\sum_{i=1}^{n}X_i^k,\quad k=1,2,\ldots
\quad\text{（反映了总体$k$阶原点矩的信息）}
\]

\textbf{$k$阶样本中心矩}：
\[
B_k=\frac{1}{n}\sum_{i=1}^{n}(X_i-\bar{X})^k,\quad k=1,2,\ldots
\quad\text{（反映了总体$k$阶中心矩的信息）}
\]

\begin{remark}
\textbf{统计量与数字特征的区别}：样本数字特征统计量是随机变量；而（随机变量的）数字特征是常数。
\end{remark}

\section{三大抽样分布}

\subsection{$\chi^2$分布}

\begin{definition}[$\chi^2$分布]
设$X_1,X_2,\ldots,X_n$是来自总体$N(0,1)$的样本，则称统计量
\[
\chi^2=\sum_{i=1}^{n}X_i^2
\]
服从自由度为$n$的\textbf{$\chi^2$分布}，记为$\chi^2\sim\chi^2(n)$。
\end{definition}

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{可加性}：设$\chi_1^2\sim\chi^2(n_1)$，$\chi_2^2\sim\chi^2(n_2)$，且$\chi_1^2$与$\chi_2^2$相互独立，则
    \[
    \chi_1^2+\chi_2^2\sim\chi^2(n_1+n_2).
    \]
    \item \textbf{数字特征}：若$\chi^2\sim\chi^2(n)$，则
    \[
    E(\chi^2)=n,\quad D(\chi^2)=2n.
    \]
\end{enumerate}

\subsection{$t$分布（Student分布）}

\begin{definition}[$t$分布]
设$X\sim N(0,1)$，$Y\sim\chi^2(n)$，且$X$与$Y$相互独立，则称随机变量
\[
t=\frac{X}{\sqrt{Y/n}}
\]
服从自由度为$n$的\textbf{$t$分布}，记为$t\sim t(n)$。
\end{definition}

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $t$分布关于$x=0$对称。
    \item $n=1$时，$E(t)$不存在；$n\geq 2$时，$E(t)=0$。
    \item $n\to\infty$时，$t$分布趋向于标准正态分布。实际应用中，当$n\geq 45$时，可用标准正态分布近似。
\end{enumerate}

\subsection{$F$分布}

\begin{definition}[$F$分布]
设$U\sim\chi^2(n_1)$，$V\sim\chi^2(n_2)$，且$U$与$V$相互独立，则称随机变量
\[
F=\frac{U/n_1}{V/n_2}
\]
服从自由度为$(n_1,n_2)$的\textbf{$F$分布}，记为$F\sim F(n_1,n_2)$。
\end{definition}

\textbf{性质}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 若$F\sim F(n_1,n_2)$，则$\dfrac{1}{F}\sim F(n_2,n_1)$。
\end{enumerate}

\section{正态总体的样本均值与样本方差的分布}

\subsection{单个正态总体}

\begin{theorem}
设$X_1,\ldots,X_n$是来自总体$X\sim N(\mu,\sigma^2)$的随机样本，则
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{样本均值的分布}：
    \[
    \bar{X}=\frac{1}{n}\sum_{i=1}^{n}X_i\sim N\left(\mu,\frac{\sigma^2}{n}\right),
    \quad\text{即}\quad
    \frac{\bar{X}-\mu}{\sigma/\sqrt{n}}\sim N(0,1).
    \]
    \item \textbf{样本方差的分布}（总体方差$\sigma^2$已知）：
    \[
    \frac{(n-1)S^2}{\sigma^2}
    =\frac{\sum_{i=1}^{n}(X_i-\bar{X})^2}{\sigma^2}
    \sim\chi^2(n-1).
    \]
    \item \textbf{独立性}：样本均值$\bar{X}$与样本方差$S^2$相互独立。
    \item \textbf{样本均值与样本方差的联合分布}（总体方差$\sigma^2$未知）：
    \[
    \frac{\bar{X}-\mu}{S/\sqrt{n}}\sim t(n-1).
    \]
\end{enumerate}
\end{theorem}

\subsection{两个正态总体}

\begin{theorem}
设$X_1,\ldots,X_{n_1}$与$Y_1,\ldots,Y_{n_2}$分别是来自正态总体$N(\mu_1,\sigma_1^2)$和$N(\mu_2,\sigma_2^2)$的随机样本，且两样本相互独立，则
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item \textbf{样本的方差比分布}：
    \[
    \frac{S_1^2/S_2^2}{\sigma_1^2/\sigma_2^2}\sim F(n_1-1,n_2-1).
    \]
    \item \textbf{样本的均值差分布}（当$\sigma_1^2=\sigma_2^2=\sigma^2$时）：
    \[
    \frac{(\bar{X}-\bar{Y})-(\mu_1-\mu_2)}{S_w\sqrt{\dfrac{1}{n_1}+\dfrac{1}{n_2}}}
    \sim t(n_1+n_2-2),
    \]
    其中
    \[
    S_w^2=\frac{(n_1-1)S_1^2+(n_2-1)S_2^2}{n_1+n_2-2},\quad
    S_w=\sqrt{S_w^2}.
    \]
\end{enumerate}
\end{theorem}

\section{分布的分位数（分位点）}

\begin{definition}[上$\alpha$分位数]
设总体$X$的分布函数为$F(x)$，密度函数为$f(x)$。对于给定正数$\alpha$（$0<\alpha<1$），满足条件
\[
P\{X>x_\alpha\}=\alpha
\quad\text{或}\quad
F(x_\alpha)=1-\alpha
\]
的数$x_\alpha$称为分布函数$F(x)$的\textbf{上侧$\alpha$分位数}（上$\alpha$分位点）。
\end{definition}

\textbf{$\chi^2$分布的分位数}：
\[
P\{\chi^2>\chi_\alpha^2(n)\}=\alpha.
\]
R.A.\ Fisher证明当$n$充分大时（$n>40$），
\[
\chi_\alpha^2(n)\approx\frac{1}{2}\left(z_\alpha+\sqrt{2n-1}\right)^2,
\]
其中$z_\alpha$是标准正态分布的上$\alpha$分位点。

\textbf{$t$分布的分位数}：
\[
P\{t>t_\alpha(n)\}=\alpha.
\]
由概率密度函数的对称性知
\[
t_{1-\alpha}(n)=-t_\alpha(n).
\]
当$n$充分大时（$n>45$），$t_\alpha(n)\approx z_\alpha$。

\textbf{$F$分布的分位数}：
\[
P\{F>F_\alpha(n_1,n_2)\}=\alpha.
\]
且有性质
\[
F_{1-\alpha}(n_1,n_2)=\frac{1}{F_\alpha(n_2,n_1)}.
\]

\section{典型例题}

\begin{example}[例1]
设$X_1,X_2,X_3,Y_1,Y_2,Y_3,Y_4$是来自正态总体$N(0,1)$的独立样本，判断下列统计量的分布：
\[
X_1^2+X_2^2+X_3^2,\qquad
\frac{4\sum_{i=1}^{3}X_i^2}{3\sum_{i=1}^{4}Y_i^2},\qquad
\frac{\sqrt{3}X_1}{\sqrt{Y_1^2+Y_2^2+Y_3^2}}.
\]

\textbf{解}：根据$\chi^2$变量、$t$变量、$F$变量的构成模式：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item $X_1^2+X_2^2+X_3^2\sim\chi^2(3)$。
    \item $\sum_{i=1}^{3}X_i^2\sim\chi^2(3)$，$\sum_{i=1}^{4}Y_i^2\sim\chi^2(4)$，且相互独立，故
    \[
    \frac{4\sum_{i=1}^{3}X_i^2}{3\sum_{i=1}^{4}Y_i^2}
    =\frac{\sum X_i^2/3}{\sum Y_i^2/4}
    \sim F(3,4).
    \]
    \item $X_1\sim N(0,1)$，$Y_1^2+Y_2^2+Y_3^2\sim\chi^2(3)$，且相互独立，故
    \[
    \frac{\sqrt{3}X_1}{\sqrt{Y_1^2+Y_2^2+Y_3^2}}
    =\frac{X_1}{\sqrt{(Y_1^2+Y_2^2+Y_3^2)/3}}
    \sim t(3).
    \]
\end{enumerate}
\end{example}

\begin{example}[例2]
设$X_1,X_2,\ldots,X_{10}$独立同分布，$X_i\sim N(0,0.3^2)$，求$c$的值，使$c\sum_{i=1}^{10}X_i^2$服从$\chi^2$分布，并求自由度$n$。

\textbf{解}：由于$X_i\sim N(0,0.3^2)$，则$\dfrac{X_i}{0.3}\sim N(0,1)$。于是
\[
\left(\frac{X_1}{0.3}\right)^2+\left(\frac{X_2}{0.3}\right)^2+\cdots+\left(\frac{X_{10}}{0.3}\right)^2
=\frac{1}{0.09}\sum_{i=1}^{10}X_i^2
\sim\chi^2(10).
\]
故$c=\dfrac{1}{0.09}=\dfrac{100}{9}$，自由度$n=10$。
\end{example}

\begin{example}[例3]
求总体$N(20,3)$的容量分别为$10,15$的两独立样本均值差的绝对值大于$0.3$的概率。

\textbf{解}：设两样本均值分别为$\bar{X},\bar{Y}$，则
\[
\bar{X}\sim N\left(20,\frac{3}{10}\right),\quad
\bar{Y}\sim N\left(20,\frac{3}{15}\right).
\]
由于两样本独立，故
\[
\bar{X}-\bar{Y}\sim N\left(0,\frac{3}{10}+\frac{3}{15}\right)=N\left(0,\frac{1}{2}\right).
\]
因此
\[
\begin{aligned}
P\{|\bar{X}-\bar{Y}|>0.3\}
&=1-P\left\{\frac{-0.3}{\sqrt{1/2}}\leq\frac{\bar{X}-\bar{Y}}{\sqrt{1/2}}\leq\frac{0.3}{\sqrt{1/2}}\right\}\\
&=1-[2\Phi(0.424)-1]\approx 0.6744.
\end{aligned}
\]
\end{example}

\begin{example}[例4]
从总体$X\sim N(5,2^2)$中随机抽取一容量为$25$的样本，求样本均值$\bar{X}$落在$4.2$到$5.8$之间的概率，以及样本方差$S^2$大于$6.07$的概率。

\textbf{解}：由于总体$X\sim N(5,4)$，$n=25$，故
\[
\bar{X}\sim N\left(5,\frac{4}{25}\right),\qquad
\frac{(25-1)S^2}{2^2}\sim\chi^2(24).
\]

样本均值落在$4.2$到$5.8$之间的概率：
\[
P\{4.2<\bar{X}<5.8\}
=P\left\{-2<\frac{\bar{X}-5}{2/5}<2\right\}
=2\Phi(2)-1=0.908.
\]

样本方差$S^2$大于$6.07$的概率：
\[
P\{S^2>6.07\}
=P\left\{\frac{24S^2}{4}>\frac{6.07\times 24}{4}\right\}
=P\{\chi^2(24)>36.42\}=0.05.
\]
其中$36.42=\chi_{0.05}^2(24)$，即自由度为$24$的$\chi^2$分布的上侧$0.05$分位数。
\end{example}

\begin{example}[例5]
设总体$X\sim N(\mu,\sigma^2)$，从总体中取一个容量为$n=16$的样本$(X_1,X_2,\ldots,X_{16})$，求：
\[
\text{(1)}\ P\left\{\frac{\sigma^2}{2}\leq\frac{1}{n}\sum_{i=1}^{n}(X_i-\mu)^2\leq 2\sigma^2\right\};
\qquad
\text{(2)}\ P\left\{\frac{\sigma^2}{2}\leq\frac{1}{n}\sum_{i=1}^{n}(X_i-\bar{X})^2\leq 2\sigma^2\right\}.
\]

\textbf{解}：(1) 因为$\sum_{i=1}^{n}\left(\dfrac{X_i-\mu}{\sigma}\right)^2\sim\chi^2(16)$，于是
\[
\begin{aligned}
P\left\{\frac{\sigma^2}{2}\leq\frac{1}{n}\sum_{i=1}^{n}(X_i-\mu)^2\leq 2\sigma^2\right\}
&=P\left\{8\leq\sum_{i=1}^{16}\frac{(X_i-\mu)^2}{\sigma^2}\leq 32\right\}\\
&=P\{8\leq\chi^2(16)\leq 32\}\\
&=P\{\chi^2(16)\leq 32\}-P\{\chi^2(16)\leq 8\}\\
&=(1-0.01)-(1-0.95)=0.94.
\end{aligned}
\]

(2) 因为$\sum_{i=1}^{n}\left(\dfrac{X_i-\bar{X}}{\sigma}\right)^2=\dfrac{(n-1)S^2}{\sigma^2}\sim\chi^2(15)$，于是
\[
\begin{aligned}
P\left\{\frac{\sigma^2}{2}\leq\frac{1}{n}\sum_{i=1}^{n}(X_i-\bar{X})^2\leq 2\sigma^2\right\}
&=P\left\{8\leq\sum_{i=1}^{16}\frac{(X_i-\bar{X})^2}{\sigma^2}\leq 32\right\}\\
&=P\{8\leq\chi^2(15)\leq 32\}\\
&=P\{\chi^2(15)\geq 8\}-P\{\chi^2(15)\geq 32\}\\
&\approx 0.92-0.005=0.915.
\end{aligned}
\]
\end{example}

\begin{example}[例6]
为了估计灯泡寿命，抽取$10$个灯泡测试。求得样品均值$\bar{X}=1500$小时，样品标准差$S=20$小时。若灯泡寿命服从正态分布$N(\mu,\sigma_0^2)$，以$\bar{X}$作为$\mu$的估计，试求：
\begin{enumerate}[label=(\arabic*),leftmargin=2em]
    \item 若已知$\sigma_0^2=400$，$\bar{X}$偏离总体均值$\mu$不超过$12$小时的概率。
    \item 若未知$\sigma_0^2$，$\bar{X}$偏离总体均值$\mu$不超过$12$小时的概率。
\end{enumerate}

\textbf{解}：(1) 已知$\sigma_0^2=400$，则$\dfrac{\bar{X}-\mu}{\sigma_0/\sqrt{n}}\sim N(0,1)$，
\[
\begin{aligned}
P\{|\bar{X}-\mu|\leq 12\}
&=P\left\{\left|\frac{\bar{X}-\mu}{20/\sqrt{10}}\right|\leq\frac{12}{20/\sqrt{10}}\right\}\\
&=P\{|Z|\leq 1.897\}=2\Phi(1.897)-1\approx 0.94.
\end{aligned}
\]

(2) 未知$\sigma_0^2$，利用$\dfrac{\bar{X}-\mu}{S/\sqrt{n}}\sim t(n-1)$，
\[
P\{|\bar{X}-\mu|\leq 12\}
=P\left\{\left|\frac{\bar{X}-\mu}{20/\sqrt{10}}\right|\leq 1.897\right\}.
\]
查表得$t_{0.05}(9)=1.8331$，由$t$分布的对称性，
\[
P\{|t(9)|\leq 1.897\}\approx 1-2\times 0.05=0.9.
\]
\end{example}

\section{Excel实验}

\subsection{常见统计分布的分位数点}

Excel中形式如\texttt{XXX.INV}的统计函数，是形如\texttt{XXX.DIST}的反函数。

\textbf{常用函数}：
\begin{itemize}[leftmargin=2em]
    \item \texttt{NORM.S.INV(probability)}：返回标准正态分布的区间点。
    \item \texttt{T.INV(probability, deg\_freedom)}：返回$t$分布的左尾区间点。
    \item \texttt{F.INV(probability, deg\_freedom1, deg\_freedom2)}：返回（左尾）$F$分布的逆函数值。
    \item \texttt{F.INV.RT(probability, deg\_freedom1, deg\_freedom2)}：返回（右尾）$F$分布的逆函数值。
    \item \texttt{CHISQ.INV(probability, deg\_freedom)}：返回具有给定概率的左尾$\chi^2$分布的区间点。
    \item \texttt{CHISQ.INV.RT(probability, deg\_freedom)}：返回具有给定概率的右尾$\chi^2$分布的区间点。
\end{itemize}

\begin{example}[实验1]
用Excel求下列分位数：
\[
z_{0.1},\quad t_{0.975}(4),\quad t_{0.05}(55),\quad F_{0.9}(14,10),\quad \chi_{0.975}^2(200).
\]

\textbf{Excel公式}：
\begin{itemize}[leftmargin=2em]
    \item \texttt{=NORM.S.INV(0.9)}
    \item \texttt{=T.INV(1-0.975, 4)}
    \item \texttt{=T.INV(1-0.05, 55)} 或 \texttt{=T.INV.2T(0.1,55)}
    \item \texttt{=F.INV(1-0.9, 14, 10)} 或 \texttt{=F.INV.RT(0.9, 14, 10)}
    \item \texttt{=CHISQ.INV.RT(0.975, 200)} 或 \texttt{=CHISQ.INV(1-0.975, 200)}
\end{itemize}
\end{example}

\subsection{使用Excel做区间估计}

\begin{example}[实验2]
已知某种灯泡的寿命服从正态分布，现从一批灯泡中抽取$16$只，测得其寿命（单位：小时）如下：
\begin{center}
\begin{tabular}{cccccccc}
1510 & 1450 & 1480 & 1460 & 1520 & 1480 & 1490 & 1460\\
1480 & 1510 & 1530 & 1470 & 1500 & 1520 & 1510 & 1470
\end{tabular}
\end{center}
求该灯泡平均使用寿命置信度为$90\%$，$95\%$，$99\%$的置信区间。

\textbf{分析}：总体方差未知，关于总体均值$\mu$的置信度为$1-\alpha$的置信区间为
\[
\left(\bar{X}\pm\frac{S}{\sqrt{n}}t_{\alpha/2}(n-1)\right).
\]

\textbf{Excel操作}：
\begin{enumerate}[label=\arabic*.,leftmargin=2em]
    \item 样本容量：\texttt{=COUNT(A2:B9)}
    \item 样本均值：\texttt{=AVERAGE(A2:B9)}
    \item 样本方差：\texttt{=VAR(A2:B9)}
    \item $t$值：\texttt{=T.INV(1-E5, E2-1)}（其中\texttt{TINV($\alpha,n$)}返回的是$t_{\alpha/2}(n)$）
    \item 置信区间下限：\texttt{=E3-SQRT(E4/E2)*E6}
    \item 置信区间上限：\texttt{=E3+SQRT(E4/E2)*E6}
\end{enumerate}
\end{example}

\begin{example}[使用Python计算置信区间]

import numpy as np
from scipy import stats

# 示例数据
data = [1510, 1450, 1480, 1460, 1520, 1480, 1490, 1460,

interval = stats.t.interval(confidence, len(data)-1, loc=mean, scale=sem)

print(f"{confidence*100}%的置信区间为: {interval}")
\end{verbatim}
\end{example}
